```Makefile
# Makefile for OpenMP Offload microXOR

CC ?= g++
SM_VERSION ?= 80
SRC = src
INC = include
OBJ = obj
CXXFLAGS ?= -fopenmp -O3 -std=c++17 -I$(INC)
ifneq ($(DEBUG), yes)
	CXXFLAGS += -O3
else
	CXXFLAGS += -g
endif
LDFLAGS ?= -L/usr/local/cuda/lib64 -lcudart

CXXFILES = $(SRC)/microXOR.cpp $(SRC)/main.cpp
DEPS = $(INC)/microXOR.hpp
OBJS = $(CXXFILES:$(SRC)/%.cpp=$(OBJ)/%.o)
EXE = microXOR.exe

$(EXE): $(OBJS)
	$(CC) $(CXXFLAGS) $(OBJS) -o $(EXE) $(LDFLAGS)

$(OBJ)/%.o: $(SRC)/%.cpp $(DEPS) | obj
	$(CC) $(CXXFLAGS) -c $< -o $@

obj:
	mkdir -p $(OBJ)

clean:
	rm -f $(OBJS) $(EXE)

.PHONY: clean
```

### Notes:
1. The file extensions have been changed from `.cu` to `.cpp` to reflect the transition from CUDA to OpenMP.
2. The `nvcc` compiler has been replaced with `g++`, which is commonly used for OpenMP.
3. The `-fopenmp` flag is added to enable OpenMP support.
4. The `LDFLAGS` line is modified to include the necessary libraries for OpenMP and CUDA runtime.
5. The `SM_VERSION` variable is retained for potential use in device-specific optimizations, although it may not be directly applicable in OpenMP.
