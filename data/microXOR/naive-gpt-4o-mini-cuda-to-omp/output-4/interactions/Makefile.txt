Here is the translated Makefile for the OpenMP Offload execution model:

```makefile
# Makefile for OpenMP Offload microXOR

CXX ?= $(CXX_COMPILER)
SM_VERSION ?= 80
SRC = src
INC = include
OBJ = obj
CXXFLAGS ?= -fopenmp -fopenmp-targets=nvptx-none -I$(INC) -std=c++17
ifneq ($(DEBUG), yes)
	CXXFLAGS += -O3
else
	CXXFLAGS += -g
endif
LDFLAGS ?= -lgomp

CPPFILES = $(SRC)/microXOR.cpp $(SRC)/main.cpp
DEPS = $(INC)/microXOR.hpp
OBJS = $(CPPFILES:$(SRC)/%.cpp=$(OBJ)/%.o)
EXE = microXOR.exe

$(EXE): $(OBJS)
	$(CXX) $(CXXFLAGS) $(OBJS) -o $(EXE) $(LDFLAGS)

$(OBJ)/%.o: $(SRC)/%.cpp $(DEPS) | obj
	$(CXX) $(CXXFLAGS) -c $< -o $@

obj:
	mkdir -p $(OBJ)

clean:
	rm -f $(OBJS) $(EXE)

.PHONY: clean
```

### Notes:
1. The file extensions have been changed from `.cu` to `.cpp` and `.cuh` to `.hpp` to reflect the C++ code style.
2. The `CXXFLAGS` now include flags for OpenMP Offload with `-fopenmp` and `-fopenmp-targets=nvptx-none`.
3. The `LDFLAGS` includes the OpenMP library `-lgomp`.
4. The `CXX` variable is used to specify the C++ compiler, which can be set by the user when invoking `make`.
