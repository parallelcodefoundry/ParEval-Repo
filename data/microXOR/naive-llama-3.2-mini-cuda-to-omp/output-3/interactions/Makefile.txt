	$(CXX) -c $(CXX_FILES) $(BUILD_FLAGS)

cellsXOR.o: cellsXOR.cpp $(header_files)

Also note that this Makefile is written for GCC 4.7 or later, which supports OpenMP 3.0 and CUDA target option (`-fopenmp --target=nvptx32`). You may need to modify the Makefile if you're using an earlier version of GCC.
