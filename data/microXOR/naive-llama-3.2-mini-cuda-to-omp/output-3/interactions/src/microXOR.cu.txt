Here's the translated version of `src/microXOR.cu` using OpenMP offloading:

```cpp
// microXOR kernel (OpenMP offloading)

#include <omp.h>
#include "microXOR.cuh"

__device__ int cellsXOR(const int *input, int *output, size_t N) {
  int i = blockIdx.y * blockDim.y + threadIdx.y;
  int j = blockIdx.x * blockDim.x + threadIdx.x;
  if (i < N && j < N) {
    int count = 0;
    if (i > 0 && input[(i-1)*N + j] == 1) count++;
    if (i < N-1 && input[(i+1)*N + j] == 1) count++;
    if (j > 0 && input[i*N + (j-1)] == 1) count++;
    if (j < N-1 && input[i*N + (j+1)] == 1) count++;
    output[i*N + j] = (count == 1) ? 1 : 0;
  }
}

int main() {
  // Initialize data
  int N = 4; // grid size

  #pragma omp parallel for reduction(+:output[0]) num_threads(N*N)
  for (size_t i = 0; i < N; i++) {
    for (size_t j = 0; j < N; j++) {
      __syncthreadid tid;
      __threadlocal int threadIdX = tid % N;
      __threadlocal int threadIdY = tid / N;

      if (threadIdX == 0 && threadIdY == 0) { // first element of the grid
        cellsXOR(&input[0], &output[0], N);
        break; // stop once one result is computed
      }
    }
  }

  return 0;
}
```

Note that OpenMP offloading is implemented on top of CUDA, and `__device__` functions are used instead of the default kernel definition. The `main` function initializes data and uses OpenMP to parallelize the computation using multiple threads.

However, since we want the OpenMP code to be executed by a host thread and not on the GPU, we need to use the following directives:

```cpp
#pragma omp target device gpu
__syncthreadid tid;
```

Here's how you can do it:

```c++
// microXOR kernel (OpenMP offloading)

#include <omp.h>
#include "microXOR.cuh"

extern "C" void cellsXORHost(int *input, int *output, size_t N) {
  #pragma omp target device gpu
  __syncthreadid tid;

  if (__omp_task_for(0, N)) return; // only execute the loop if this thread is not in the last iteration

  for (size_t i = 0; i < N; i++) {
    for (size_t j = 0; j < N; j++) {
      __threadlocal int threadIdX = tid % N;
      __threadlocal int threadIdY = tid / N;

      if (threadIdX == 0 && threadIdY == 0) { // first element of the grid
        int count = 0;
        if (i > 0 && input[(i-1)*N + j] == 1) count++;
        if (i < N-1 && input[(i+1)*N + j] == 1) count++;
        if (j > 0 && input[i*N + (j-1)] == 1) count++;
        if (j < N-1 && input[i*N + (j+1)] == 1) count++;
        output[i*N + j] = (count == 1) ? 1 : 0;
      }
    }
  }
}

int main() {
  // Initialize data
  int N = 4; // grid size

  int *input, *output;
  cudaMalloc((void **)&input, N * N * sizeof(int));
  cudaMalloc((void **)&output, N * N * sizeof(int));

  cudaMemcpy(input, nullptr, N * N * sizeof(int), cudaMemcpyHostToDevice);

  cellsXORHost(input, output, N);

  cudaMemcpy(output, input, N * N * sizeof(int), cudaMemcpyDeviceToHost);
  cudaFree(input);
  cudaFree(output);

  return 0;
}
```

Please note that OpenMP offloading is still an experimental feature and not fully supported by all compilers.
