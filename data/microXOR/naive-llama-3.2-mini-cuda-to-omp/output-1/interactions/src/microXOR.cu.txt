#include <omp.h>
#include "microXOR.cuh"

#define N 1024

__host__ void cellsXOR(const int *input, int *output) {
  size_t i, j;
  for (i = 0; i < N; ++i) {
    for (j = 0; j < N; ++j) {
      int count = 0;
      if (i > 0 && input[(i-1)*N + j] == 1) count++;
      if (i < N-1 && input[(i+1)*N + j] == 1) count++;
      if (j > 0 && input[i*N + (j-1)] == 1) count++;
      if (j < N-1 && input[i*N + (j+1)] == 1) count++;
      output[i*N + j] = (count == 1) ? 1 : 0;
    }
  }

  // Offload to GPU

    }

The offloaded function is marked with `__device__` as it should be executed on the device (GPU). However, since this function cannot be inlined or used directly in a kernel launch, we use OpenMP-Offload's `offload` keyword with `target(gpu)` and `no_data_copy` attributes to offload its body.
