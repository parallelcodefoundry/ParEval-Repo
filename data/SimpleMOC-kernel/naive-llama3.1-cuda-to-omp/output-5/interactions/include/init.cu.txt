// Build exponential table function
void buildExponentialTable(Table* table, int num_threads) {
    // Define table values on device
    float* values = (float*) malloc(2 * num_sources * sizeof(float));
    for (int i = 0; i < 2 * num_sources; i++) {
        values[i] = 0.0f;
    }

    // Copy data to device
    cudaMemcpy(values, table->values_host, 2 * num_sources * sizeof(float), cudaMemcpyHostToDevice);

    // Perform computation on device
    #pragma omp target map(to:values) device(0)
    {
        computeExponentialTable(table);
    }
}

// Compute exponential table function
void computeExponentialTable(Table* table) {
    // Define number of elements in array
    int N = table->N;

    // Perform computation
    for (int n = 0; n < N; n++) {
        float dx = table->dx;
        float exponential = exp(-n * dx);
        table->values[2*n] = -exponential;
        table->values[2*n+1] = 1 + (n * dx - 1) * exponential;
    }
}
