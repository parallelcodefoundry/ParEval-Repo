Here is the translated version of `include/main.cu` from CUDA to OpenMP Offload:

```c
#include "SimpleMOC-header.h"
#include <omp.h>

int main(int argc, char *argv[]) {
    int version = 4;

    srand(time(NULL));

    Input I = set_default_input();
    read_CLI(argc, argv, &I);

    // Calculate Number of 3D Source Regions
    I.source_3D_regions = (int) ceil((double) I.source_2D_regions *
        I.coarse_axial_intervals / I.decomp_assemblies_ax);

    logo(version);

    print_input_summary(I);

    center_print("INITIALIZATION", 79);
    border_print();

    // Build Source Data
    printf("Building Source Data Arrays...\n");
    Source_Arrays SA_h, SA_d;
    Source *sources_h = initialize_sources(I, &SA_h);
    Source *sources_d = initialize_device_sources(I, &SA_h, &SA_d, sources_h);

    // Build Exponential Table
#ifdef TABLE
    printf("Building Exponential Table...\n");
    Table table = buildExponentialTable();
#endif

    // Setup OpenMP threads
#pragma omp parallel for num_threads(I.nthreads)
    for (int i = 0; i < I.streams; i++) {
        curandState state;
        curand_init(1234, i, 0, &state);
    }

    // Allocate Some Flux State vectors to randomly pick from
    printf("Setting up Flux State Vectors...\n");
    float *flux_states;
    int N_flux_states = 10000;
    assert(I.segments >= N_flux_states);
#ifdef OFFLOAD
    flux_states = (float *) omp_offload_alloc(N_flux_states * I.egroups * sizeof(float));
#endif

    // Run Simulation Kernel Loop
    for (int i = 0; i < I.segments / I.seg_per_thread; i++) {
        int blockId = i;

#pragma offload target(mic:MIC_1) in(I, SA_d, table)
        {
            run_kernel<<<(I.egroups + 255)/256, I.egroups>>>(I, sources_d, SA_d, table, flux_states, N_flux_states);
            cudaDeviceSynchronize();
        }

#pragma offload target(mic:MIC_1) in(SA_h) out(flux_states)
        {
            // Copy data from device to host
            cudaMemcpy(flux_states, SA_h.fine_source_arr, N_flux_states * I.egroups * sizeof(float), cudaMemcpyDeviceToHost);
#ifdef TABLE
            // Free memory allocated for exponential table
            free(table.values);
#endif
        }

#pragma offload target(mic:MIC_1) in(SA_d)
        {
            // Free memory allocated on device
            free(sources_d);
            free(SA_d.fine_source_arr);
            free(SA_d.fine_flux_arr);
            free(SA_d.sigT_arr);
#ifdef TABLE
            free(table.values);
#endif
        }
    }

    printf("Initialization Complete.\n");
    border_print();
    center_print("SIMULATION", 79);
    border_print();

    return 0;
}
```

This code uses OpenMP's offloading features to move kernel execution and data management onto an NVIDIA GPU, similar to how CUDA works. Please note that you may need to modify the `OFFLOAD` directive to point to your specific device.

Also, this is just one possible translation of the original CUDA code to OpenMP Offload, there might be other ways to achieve the same result depending on your specific use case and requirements.
