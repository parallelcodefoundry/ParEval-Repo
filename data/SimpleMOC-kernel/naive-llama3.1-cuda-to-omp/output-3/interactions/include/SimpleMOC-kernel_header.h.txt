#ifndef __SimpleMOC_header
#define __SimpleMOC_header

#include <omp.h>
#include <openacc.h>

// User inputs

// Source Arrays
typedef struct{
	float * fine_flux_arr;

	float * sigT_arr;

	float values[706];

	int N;

Note that the translation involves:

* Including `omp.h` and `openacc.h` for OpenMP-Offload support.
* Defining a custom region with `#pragma acc routine seq` directives to specify which functions should be offloaded.
* Adding the `__device__` keyword to function declarations, which indicates they can be executed on the device.
* Using OpenACC pragmas (`#pragma acc routine`) to specify which functions should be offloaded.
