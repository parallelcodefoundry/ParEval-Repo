#include "SimpleMOC-kernel_header.h"
#include <omp.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>
#include <time.h>
#include <assert.h>
#include <pthread.h>
#include <unistd.h>
#include <malloc.h>

// Function to initialize the kernel
void run_kernel(Input I, Source *S, Source_Arrays SA, Table *table, float *state_fluxes, int N_state_fluxes) {
    int blockId, g, i;
    float q0, q1, q2, sigT, tau, sigT2, expVal, reuse, flux_integral, tally, t1, t2, t3, t4;
    float dz = 0.1f, zin = 0.3f, weight = 0.5f, mu = 0.9f, mu2 = 0.3f, ds = 0.7f;
    const int egroups = I.egroups;
    int *state_flux_id, *QSR_id, *FAI_id;
    float *FSR_flux, *state_flux, *f1, *f2, *f3;

    #pragma omp target teams distribute parallel for private(blockId, g, q0, q1, q2, sigT, tau, sigT2, expVal, reuse, flux_integral, tally, t1, t2, t3, t4, dz, zin, weight, mu, mu2, ds, state_flux_id, QSR_id, FAI_id, FSR_flux, state_flux, f1, f2, f3)
    for (blockId = 0; blockId < I.segments / I.seg_per_thread; blockId++) {
        state_flux_id = (int *)malloc(I.seg_per_thread * sizeof(int));
        QSR_id = (int *)malloc(I.seg_per_thread * sizeof(int));
        FAI_id = (int *)malloc(I.seg_per_thread * sizeof(int));

        for (i = 0; i < I.seg_per_thread; i++) {
            state_flux_id[i] = rand() % N_state_fluxes;
            QSR_id[i] = rand() % I.source_3D_regions;
            FAI_id[i] = rand() % I.fine_axial_intervals;
        }

        for (i = 0; i < I.seg_per_thread; i++) {
            state_flux = &state_fluxes[state_flux_id[i] * I.egroups];

            // Attenuate Segment
            if (FAI_id[i] == 0) {
                f2 = &SA.fine_source_arr[S[QSR_id[i]].fine_source_id + (FAI_id[i]) * egroups];
                f3 = &SA.fine_source_arr[S[QSR_id[i]].fine_source_id + (FAI_id[i] + 1) * egroups];
                // cycle over energy groups
                // load neighboring sources
                for (g = 0; g < egroups; g++) {
                    float y2 = f2[g];
                    float y3 = f3[g];

                    // do linear "fitting"
                    float c0 = y2;
                    float c1 = (y3 - y2) / dz;

                    // calculate q0, q1, q2
                    q0 = c0 + c1 * zin;
                    q1 = c1;
                    q2 = 0;
                }
            } else if (FAI_id[i] == I.fine_axial_intervals - 1) {
                f1 = &SA.fine_source_arr[S[QSR_id[i]].fine_source_id + (FAI_id[i] - 1) * egroups];
                f2 = &SA.fine_source_arr[S[QSR_id[i]].fine_source_id + (FAI_id[i]) * egroups];
                // cycle over energy groups
                // load neighboring sources
                for (g = 0; g < egroups; g++) {
                    float y1 = f1[g];
                    float y2 = f2[g];

                    // do linear "fitting"
                    float c0 = y2;
                    float c1 = (y2 - y1) / dz;

                    // calculate q0, q1, q2
                    q0 = c0 + c1 * zin;
                    q1 = c1;
                    q2 = 0;
                }
            } else {
                f1 = &SA.fine_source_arr[S[QSR_id[i]].fine_source_id + (FAI_id[i] - 1) * egroups];
                f2 = &SA.fine_source_arr[S[QSR_id[i]].fine_source_id + (FAI_id[i]) * egroups];
                f3 = &SA.fine_source_arr[S[QSR_id[i]].fine_source_id + (FAI_id[i] + 1) * egroups];
                // cycle over energy groups
                // load neighboring sources
                for (g = 0; g < egroups; g++) {
                    float y1 = f1[g];
                    float y2 = f2[g];
                    float y3 = f3[g];

                    // do quadratic "fitting"
                    float c0 = y2;
                    float c1 = (y1 - y3) / (2.0f * dz);
                    float c2 = (y1 - 2.0f * y2 + y3) / (2.0f * dz * dz);

                    // calculate q0, q1, q2
                    q0 = c0 + c1 * zin + c2 * zin * zin;
                    q1 = c1 + 2.0f * c2 * zin;
                    q2 = c2;
                }
            }

            // load total cross section
            sigT = SA.sigT_arr[S[QSR_id[i]].sigT_id];

            // calculate common values for efficiency
            tau = sigT * ds;
            sigT2 = sigT * sigT;

            #ifdef TABLE
            interpolateTable(table, tau, &expVal);
            #else
            expVal = 1.0f - expf(-tau); // EXP function is faster than table lookup
            #endif

            // Flux Integral

            // Re-used Term
            reuse = tau * (tau - 2.0f) + 2.0f * expVal / (sigT * sigT2);

            // add contribution to new source flux
            for (g = 0; g < egroups; g++) {
                flux_integral = (q0 * tau + (sigT * state_flux[g] - q0) * expVal) / sigT2 + q1 * mu * reuse + q2 * mu2 * (tau * (tau * (tau - 3.0f) + 6.0f) - 6.0f * expVal) / (3.0f * sigT2 * sigT2);

                // Prepare tally
                tally = weight * flux_integral;

                // SHOULD BE ATOMIC HERE!
                //FSR_flux[g] += tally;
                FSR_flux = &SA.fine_flux_arr[S[QSR_id[i]].fine_flux_id + FAI_id[i] * egroups];
                FSR_flux[g] += tally;

                // Term 1
                t1 = q0 * expVal / sigT;
                // Term 2
                t2 = q1 * mu * (tau - expVal) / sigT2;
                // Term 3
                t3 = q2 * mu2 * reuse;
                // Term 4
                t4 = state_flux[g] * (1.0f - expVal);
                // Total psi
                state_flux[g] = t1 + t2 + t3 + t4;
            }
        }

        free(state_flux_id);
        free(QSR_id);
        free(FAI_id);
    }
}

int main(int argc, char *argv[]) {
    int version = 4;

    srand(time(NULL));

    Input I = set_default_input();
    read_CLI(argc, argv, &I);

    logo(version);

    print_input_summary(I);

    printf("INITIALIZATION\n");
    printf("Building Source Data Arrays...\n");
    Source_Arrays SA_h, SA_d;
    Source *sources_h = initialize_sources(I, &SA_h);
    Source *sources_d = initialize_device_sources(I, &SA_h, &SA_d, sources_h);

    Table *table_d = NULL;
    #ifdef TABLE
    printf("Building Exponential Table...\n");
    Table table = buildExponentialTable();
    #endif

    printf("Setting up CUDA RNG...\n");
    float *flux_states;
    int N_flux_states = 10000;
    assert(I.segments >= N_flux_states);
    flux_states = (float *)malloc(N_flux_states * I.egroups * sizeof(float));
    init_flux_states(flux_states, N_flux_states, I);

    printf("Initialization Complete.\n");
    printf("SIMULATION\n");
    printf("Attentuating fluxes across segments...\n");

    run_kernel(I, sources_h, SA_h, table_d, flux_states, N_flux_states);

    printf("Simulation Complete.\n");

    printf("RESULTS SUMMARY\n");

    return 0;
}
