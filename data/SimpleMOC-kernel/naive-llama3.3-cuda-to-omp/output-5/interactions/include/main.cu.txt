#include "SimpleMOC-kernel_header.h"
#include <omp.h>

int main(int argc, char *argv[]) {
    int version = 4;

    srand(time(NULL));

    Input I = set_default_input();
    read_CLI(argc, argv, &I);

    logo(version);

    print_input_summary(I);

    printf("INITIALIZATION\n");
    printf("Building Source Data Arrays...\n");
    Source_Arrays SA_h, SA_d;
    Source *sources_h = initialize_sources(I, &SA_h);
    Source *sources_d = initialize_device_sources(I, &SA_h, &SA_d, sources_h);

    Table *table_d = NULL;
#ifdef TABLE
    printf("Building Exponential Table...\n");
    Table table = buildExponentialTable();
    table_d = (Table *)malloc(sizeof(Table));
    *table_d = table;
#endif

    printf("Setting up OpenMP offload...\n");
    int n_blocks = sqrt(I.segments);
    int n_threads = I.egroups;

    printf("Setting up Flux State Vectors...\n");
    float *flux_states = (float *)malloc(I.egroups * 10000 * sizeof(float));
    init_flux_states(flux_states, 10000, I);

    printf("Initialization Complete.\n");
    printf("SIMULATION\n");
    printf("Attentuating fluxes across segments...\n");

    double start_time = omp_get_wtime();
#pragma omp target teams distribute parallel for num_teams(n_blocks) num_threads(n_threads) \
    map(to: I, sources_d[0:I.source_3D_regions], SA_d.fine_flux_arr[0:SA_d.fine_flux_arr_size], \
        SA_d.fine_source_arr[0:SA_d.fine_source_arr_size], SA_d.sigT_arr[0:SA_d.sigT_arr_size], \
        table_d[0:1], flux_states[0:10000*I.egroups]) \
    map(from: flux_states[0:10000*I.egroups])
    for (int i = 0; i < I.segments / I.seg_per_thread; i++) {
        run_kernel(I, sources_d, SA_d, table_d, flux_states, 10000, i);
    }
    double end_time = omp_get_wtime();

    printf("Simulation Complete.\n");
    printf("RESULTS SUMMARY\n");
    printf("Runtime: %.3f seconds\n", end_time - start_time);
    double tpi = ((end_time - start_time) / (double)I.segments / (double)I.egroups) * 1.0e9;
    printf("Time per Intersection: %.8lf ns\n", tpi);

    return 0;
}
