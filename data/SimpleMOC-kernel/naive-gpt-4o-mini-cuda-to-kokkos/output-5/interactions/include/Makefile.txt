Here is the translated `include/CMakeLists.txt` file for compiling the code as a Kokkos application:

```cmake
# CMakeLists.txt for SimpleMOC-kernel using Kokkos

cmake_minimum_required(VERSION 3.10)

# Project Name
project(SimpleMOC-kernel)

# Find Kokkos
find_package(Kokkos REQUIRED)

# Set the C++ standard
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Specify the source files
set(SOURCES
    main.cpp
    kernel.cpp
    init.cpp
    io.cpp
)

# Create the executable
add_executable(${PROJECT_NAME} ${SOURCES})

# Link Kokkos to the executable
target_link_libraries(${PROJECT_NAME} Kokkos::kokkos)

# Include directories
target_include_directories(${PROJECT_NAME} PRIVATE ${CMAKE_CURRENT_SOURCE_DIR})

# Set Kokkos options
if (Kokkos_ENABLE_CUDA)
    target_compile_definitions(${PROJECT_NAME} PRIVATE KOKKOS_ENABLE_CUDA)
endif()

# Optional: Set optimization flags
if (CMAKE_BUILD_TYPE MATCHES Release)
    target_compile_options(${PROJECT_NAME} PRIVATE -O3)
endif()

# Optional: Add debugging flags
if (CMAKE_BUILD_TYPE MATCHES Debug)
    target_compile_options(${PROJECT_NAME} PRIVATE -g)
endif()

# Optional: Add profiling flags
if (CMAKE_BUILD_TYPE MATCHES RelWithDebInfo)
    target_compile_options(${PROJECT_NAME} PRIVATE -pg)
endif()
```

### Instructions for Building
To compile this code using the Kokkos execution model, navigate to the `include` directory and run the following commands:

```bash
cd include/
cmake -DKOKKOS_BACKEND=CUDA -DCMAKE_CXX_COMPILER=g++ -Bbuild .
cmake --build build/
```

Make sure that Kokkos and CUDA are properly installed and available in your environment through `$PATH` and `$LD_LIBRARY_PATH`.
