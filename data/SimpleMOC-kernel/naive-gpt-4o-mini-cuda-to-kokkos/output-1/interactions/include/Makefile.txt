# CMakeLists.txt for SimpleMOC-kernel

cmake_minimum_required(VERSION 3.10)

project(SimpleMOC-kernel)

    io.cpp

if (CMAKE_BUILD_TYPE MATCHES RelWithDebInfo)

### Instructions for Building the Code
To compile this code using the Kokkos execution model, navigate to the `include/` directory and run the following commands:

```bash
cd include/
cmake -DKOKKOS_BACKEND=CUDA -DCMAKE_CXX_COMPILER=g++ -Bbuild .
cmake --build build/
```

Make sure that Kokkos and CUDA are installed and available in your environment through `$PATH` and `$LD_LIBRARY_PATH`.
