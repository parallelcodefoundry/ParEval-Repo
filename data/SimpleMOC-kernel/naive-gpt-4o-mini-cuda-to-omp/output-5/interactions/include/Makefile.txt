endif

ifeq ($(PROFILE),yes)

endif

ifeq ($(OPTIMIZE),yes)
	CFLAGS += -O3
endif

# Exponential Table Build
ifeq ($(TABLE), yes)
	CFLAGS += -DTABLE
endif

#===============================================================================
# Targets to Build
#===============================================================================

$(program): $(obj) SimpleMOC-kernel_header.h
	$(CXX) $(CFLAGS) $(obj) -o $@ $(LDFLAGS)

%.o: %.c
	$(CXX) $(CFLAGS) -c $< -o $@

clean:
	rm -rf $(program) $(obj)

	vim -p $(source) SimpleMOC-kernel_header.h
