#include "SimpleMOC-kernel_header.hpp"
#include <Kokkos_Core.hpp>

//Structure to hold Kokkos::View data for passing to kernel
struct KokkosData{
    Kokkos::View<Source*> sources;
    Kokkos::View<float*> fine_flux_arr;
    Kokkos::View<float*> fine_source_arr;
    Kokkos::View<float*> sigT_arr;
    Kokkos::View<float*> state_fluxes;

    KokkosData(Input I) : 
        sources("sources",I.source_3D_regions),
        fine_flux_arr("fine_flux_arr",I.source_3D_regions * I.fine_axial_intervals * I.egroups),
        fine_source_arr("fine_source_arr",I.source_3D_regions * I.fine_axial_intervals * I.egroups),
        sigT_arr("sigT_arr",I.source_3D_regions * I.egroups),
        state_fluxes("state_fluxes",10000 * I.egroups)
    {}
};


int main( int argc, char * argv[] )
{
	int version = 4;

	srand(time(NULL));

	Input I = set_default_input();
	read_CLI( argc, argv, &I );
	
	// Calculate Number of 3D Source Regions
	I.source_3D_regions = (int) ceil((double)I.source_2D_regions *
		I.coarse_axial_intervals / I.decomp_assemblies_ax);

	logo(version);

	print_input_summary(I);
	
	center_print("INITIALIZATION", 79);
	border_print();

    //Kokkos Initialization
    Kokkos::initialize(argc, argv);

	// Build Source Data
	printf("Building Source Data Arrays...\n");
	Source_Arrays SA_h;
    KokkosData kokkos_data(I);
	Source * sources_h = initialize_sources(I, &SA_h); 

    //Copy host data to Kokkos Views
    Kokkos::deep_copy(kokkos_data.sources, sources_h);
    Kokkos::deep_copy(kokkos_data.fine_flux_arr, SA_h.fine_flux_arr);
    Kokkos::deep_copy(kokkos_data.fine_source_arr, SA_h.fine_source_arr);
    Kokkos::deep_copy(kokkos_data.sigT_arr, SA_h.sigT_arr);


	// Build Exponential Table
	Table table_h = buildExponentialTable();
	Kokkos::View<Table> table("table",1);
    Kokkos::deep_copy(table, table_h);

	// Setup Kokkos execution space
    Kokkos::Cuda::execution_space exec_space;

	// Setup Kokkos RNG  (Implementation needed - replace with actual Kokkos RNG)
	printf("Setting up Kokkos RNG...\n");
    // ... Kokkos RNG setup ...

	// Allocate Some Flux State vectors to randomly pick from
	printf("Setting up Flux State Vectors...\n");
    //Kokkos View already allocated in KokkosData struct

	init_flux_states( kokkos_data.state_fluxes, 10000, I, exec_space );


	printf("Initialization Complete.\n");
	border_print();
	center_print("SIMULATION", 79);
	border_print();
	printf("Attentuating fluxes across segments...\n");

	// Kokkos timer variables
	Kokkos::Timer timer;
	
	// Setup kernel call parameters
	assert( I.segments % I.seg_per_thread == 0 );
    int n_blocks = I.segments / I.seg_per_thread;

	// Run Simulation Kernel Loop
	timer.reset();
	run_kernel(I, kokkos_data, table, exec_space, n_blocks);
	double time = timer.seconds();


	// Copy results back to host (If needed)
    // ... Kokkos deep copy back to host ...

	printf("Simulation Complete.\n");

	border_print();
	center_print("RESULTS SUMMARY", 79);
	border_print();

	double tpi = ((double) (time) /
			(double)I.segments / (double) I.egroups) * 1.0e9;
	printf("%-25s%.3f seconds\n", "Runtime:", time);
	printf("%-25s%.8lf ns\n", "Time per Intersection:", tpi);
	border_print();

    Kokkos::finalize();
	return 0;
}
