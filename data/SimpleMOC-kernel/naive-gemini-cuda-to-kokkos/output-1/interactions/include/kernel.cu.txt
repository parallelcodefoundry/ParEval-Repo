#include "SimpleMOC-kernel_header.hpp"

/* My parallelization scheme here is to basically have a single
 * block be a geometrical segment, with each thread within the
 * block represent a single energy phase. On the CPU, the
 * inner SIMD-ized loop is over energy (i.e, 100 energy groups).
 * This should allow for each BLOCK to have:
 * 		- A single state variable for the RNG
 * 		- A set of __shared__ SIMD vectors, each thread id being its idx
 */

KOKKOS_FUNCTION
void run_kernel_impl(const Input& I, const Source* S,
                     const Source_Arrays& SA, const Table* table,
                     Kokkos::Random_XorShift64_Pool<Kokkos::DefaultExecutionSpace> &rng_pool,
                     float* state_fluxes, int N_state_fluxes, int blockId) {

    // Assign RNG state
    Kokkos::Random_XorShift64 generator = rng_pool.get_state();

    blockId--; // Adjust blockId to be 0-indexed

    for (int i = 0; i < I.seg_per_thread; i++) {
        int segment_index = blockId + i;
        if (segment_index >= I.segments) break;


        int state_flux_id = generator() % N_state_fluxes;
        int QSR_id = generator() % I.source_3D_regions;
        int FAI_id = generator() % I.fine_axial_intervals;

        float* state_flux = &state_fluxes[state_flux_id * I.egroups];

        //////////////////////////////////////////////////////////
        // Attenuate Segment
        //////////////////////////////////////////////////////////

        // Some placeholder constants - In the full app some of these are
        // calculated based off position in geometry. This treatment
        // shaves off a few FLOPS, but is not significant compared to the
        // rest of the function.
        float dz = 0.1f;
        float zin = 0.3f;
        float weight = 0.5f;
        float mu = 0.9f;
        float mu2 = 0.3f;
        float ds = 0.7f;

        // load fine source region flux vector
        float* FSR_flux = &SA.fine_flux_arr[S[QSR_id].fine_flux_id + FAI_id * I.egroups];


        for (int g = 0; g < I.egroups; g++) {
            float q0, q1, q2, sigT, tau, sigT2, expVal, reuse, flux_integral, tally, t1, t2, t3, t4;

            if (FAI_id == 0) {
                float* f2 = &SA.fine_source_arr[S[QSR_id].fine_source_id + (FAI_id) * I.egroups];
                float* f3 = &SA.fine_source_arr[S[QSR_id].fine_source_id + (FAI_id + 1) * I.egroups];
                float y2 = f2[g];
                float y3 = f3[g];
                float c0 = y2;
                float c1 = (y3 - y2) / dz;
                q0 = c0 + c1 * zin;
                q1 = c1;
                q2 = 0;
            } else if (FAI_id == I.fine_axial_intervals - 1) {
                float* f1 = &SA.fine_source_arr[S[QSR_id].fine_source_id + (FAI_id - 1) * I.egroups];
                float* f2 = &SA.fine_source_arr[S[QSR_id].fine_source_id + (FAI_id) * I.egroups];
                float y1 = f1[g];
                float y2 = f2[g];
                float c0 = y2;
                float c1 = (y2 - y1) / dz;
                q0 = c0 + c1 * zin;
                q1 = c1;
                q2 = 0;
            } else {
                float* f1 = &SA.fine_source_arr[S[QSR_id].fine_source_id + (FAI_id - 1) * I.egroups];
                float* f2 = &SA.fine_source_arr[S[QSR_id].fine_source_id + (FAI_id) * I.egroups];
                float* f3 = &SA.fine_source_arr[S[QSR_id].fine_source_id + (FAI_id + 1) * I.egroups];
                float y1 = f1[g];
                float y2 = f2[g];
                float y3 = f3[g];
                float c0 = y2;
                float c1 = (y1 - y3) / (2.f * dz);
                float c2 = (y1 - 2.f * y2 + y3) / (2.f * dz * dz);
                q0 = c0 + c1 * zin + c2 * zin * zin;
                q1 = c1 + 2.f * c2 * zin;
                q2 = c2;
            }

            sigT = SA.sigT_arr[S[QSR_id].sigT_id + g];
            tau = sigT * ds;
            sigT2 = sigT * sigT;

            #ifdef TABLE
            interpolateTable(table, tau, &expVal);
            #else
            expVal = 1.f - expf(-tau);
            #endif

            reuse = tau * (tau - 2.f) + 2.f * expVal / (sigT * sigT2);
            flux_integral = (q0 * tau + (sigT * state_flux[g] - q0) * expVal) / sigT2 + q1 * mu * reuse + q2 * mu2 * (tau * (tau * (tau - 3.f) + 6.f) - 6.f * expVal) / (3.f * sigT2 * sigT2);
            tally = weight * flux_integral;
            Kokkos::atomic_add(&(FSR_flux[g]), tally);

            t1 = q0 * expVal / sigT;
            t2 = q1 * mu * (tau - expVal) / sigT2;
            t3 = q2 * mu2 * reuse;
            t4 = state_flux[g] * (1.f - expVal);
            state_flux[g] = t1 + t2 + t3 + t4;
        }
    }
    rng_pool.free_state(generator);
}

void run_kernel(const Input& I, const Source* S,
                const Source_Arrays& SA, const Table* table,
                Kokkos::Random_XorShift64_Pool<Kokkos::DefaultExecutionSpace> &rng_pool,
                float* state_fluxes, int N_state_fluxes) {

    int num_segments_per_block = I.segments / I.seg_per_thread;


    Kokkos::parallel_for(Kokkos::RangePolicy<Kokkos::DefaultExecutionSpace>(0, num_segments_per_block),
                         KOKKOS_LAMBDA(const int& i) {
                             run_kernel_impl(I, S, SA, table, rng_pool, state_fluxes, N_state_fluxes, i);
                         });
}


KOKKOS_FUNCTION
void interpolateTable(const Table* table, float x, float* out) {
    if (x > table->maxVal)
        *out = 1.0f;
    else {
        int interval = (int)(x / table->dx + 0.5f * table->dx);
        interval = interval * 2;
        float slope = table->values[interval];
        float intercept = table->values[interval + 1];
        float val = slope * x + intercept;
        *out = val;
    }
}
