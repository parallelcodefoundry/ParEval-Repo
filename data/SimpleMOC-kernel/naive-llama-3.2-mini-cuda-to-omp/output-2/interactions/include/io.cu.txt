#include <stdio.h>
#include <stdlib.h>
#include <math.h>

#define MAX_VAL 10.0

// Function to build a table of exponential values for linear interpolation
Table buildExponentialTable() {
    // Compute number of array values
    int N = (int) (MAX_VAL * sqrt(1.0 / (8.0 * 0.01)));

    // Compute spacing
    float dx = MAX_VAL / (float)N;

    // Store linear segment information (slope and y-intercept)
    Table table;
    for (int n = 0; n < N; n++) {
        // Compute slope and y-intercept for (1 - exp(-x))
        float exponential = exp(-n * dx);
        table.values[2*n] = -exponential;
        table.values[2*n+1] = 1 + (n * dx - 1) * exponential;
    }

    // Assign data to table
    table.dx = dx;
    table.maxVal = MAX_VAL - table.dx;
    table.N = N;

    return table;
}

Source* initialize_device_sources(Input I) {
    // Allocate memory for source array data
    Source* sources_d = (Source*)malloc(I.source_3D_regions * sizeof(Source));

    // Initialize fine source and flux values
    long N_fine = I.source_3D_regions * I.fine_axial_intervals * I.egroups;

        sources_d[i].fine_source_id = rand() % N_fine;

}
