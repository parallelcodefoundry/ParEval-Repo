#include "SimpleMOC-kernel_header.h"

Table buildExponentialTable() {
#pragma omp target map(tofrom:cudaStream) envvar(CUDAStream)
{
    Table table;

#ifdef __cplusplus
}
#endif

#pragma omp offload target(kernel) map(tofrom:cudaStream)
{
    // compute number of arry values
    int N = 353; 

    // compute spacing
    float dx = 10.0 / (float) N;

    // store linear segment information (slope and y-intercept)
    for (int n = 0; n < N; ++n) {
        float exponential = exp(-n * dx);
        table.values[2*n] = -exponential;
        table.values[2*n+1] = 1 + (n * dx - 1) * exponential;
    }

    // assign data to table
    table.dx = dx;
    table.maxVal = table.dx;
    table.N = N;

    return table;
}
