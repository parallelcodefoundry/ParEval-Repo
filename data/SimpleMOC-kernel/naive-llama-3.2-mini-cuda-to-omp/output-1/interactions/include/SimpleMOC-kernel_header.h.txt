void translateHeader() {
    #include "SimpleMOC-kernel-header.h"
    
    Table table = buildExponentialTable();
    
    Source_Arrays SA_h = {NULL, NULL, NULL};
    Source_Arrays_Device SA_d;
    Source *sources_h;
    // Allocate memory for sources_h here if necessary
    
    SA_d.fine_source_arr = (float *)malloc(table.N * sizeof(float));
    SA_d.fine_flux_arr = (float *)malloc(table.N * sizeof(float));
    SA_d.sigT_arr = (float *)malloc(table.N * sizeof(float));
    
    // Initialize tables and arrays
    for(int n = 0; n < table.N; n++) {
        SA_d.fine_source_arr[n] = -exp(-n*table.dx);
        SA_d.fine_flux_arr[n] = 1 + (n*table.dx - 1)*exp(-n*table.dx);
        SA_d.sigT_arr[n] = exp(-n*table.dx);
    }
    
    // Copy from host to device
    cudaMalloc((void **) &SA_d.sources_d, sizeof(Source));
    cudaMemcpy(SA_d.sources_d, sources_h, sizeof(Source), cudaMemcpyHostToDevice);

    printf("Table dx: %.2f\n", table.dx);
    printf("Table maxVal: %.2f\n", table.maxVal);
    printf("Table N: %d\n", table.N);
}
