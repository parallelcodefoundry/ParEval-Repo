// Function to build exponential table for linear interpolation
Table buildExponentialTable() {
    // Define table
    Table table;

    // Compute number of arry values
    int N = 353; 

    // Compute spacing
    float dx = 1.0 / (float)N;

    // Store linear segment information (slope and y-intercept)
    for(int n = 0; n < N; n++) {
        // Compute slope and y-intercept for ( 1 - exp(-x) )
        table.values[2*n] = -exp(-n * dx);
        table.values[2*n+1] = 1 + (n * dx - 1) * table.values[2*n];
    }

    // Assign data to table
    table.dx = dx;
    table.maxVal = 0.01;  
    table.N = N;

    return table;
}

// Function to manage source arrays
void manageSourceArrays(Source_Arrays* sources_h, Source_Arrays* sources_d) {
    long N_fine = sources_h->fine_source_arr ? sources_h->fine_source_arr[0] : 0;
    long N_sigT = sources_h->sigT_arr ? sources_h->sigT_arr[0] : 0;

    if(N_fine > 0 && N_sigT > 0) {
        // Copy source data to GPU
        cudaMemcpy(sources_d->fine_source_arr, sources_h->fine_source_arr, N_fine * sizeof(float), cudaMemcpyHostToDevice);
        cudaMemcpy(sources_d->fine_flux_arr, sources_h->fine_flux_arr, N_fine * sizeof(float), cudaMemcpyHostToDevice);
        cudaMemcpy(sources_d->sigT_arr, sources_h->sigT_arr, N_sigT * sizeof(float), cudaMemcpyHostToDevice);

        // Manage source arrays on GPU
        #pragma omp target map(inout: sources_d->fine_source_arr, sources_d->fine_flux_arr, sources_d->sigT_arr)
        {
            // Perform work on source arrays here
            for (long i = 0; i < N_fine; i++) {
                sources_d->fine_source_arr[i] = exp(-i * dx);
                sources_d->fine_flux_arr[i] = 1 + (i * dx - 1) * sources_d->fine_source_arr[i];
            }

            for (long i = 0; i < N_sigT; i++) {

            }
        }

        // Copy source data from GPU back to host
        cudaMemcpy(sources_h->fine_source_arr, sources_d->fine_source_arr, N_fine * sizeof(float), cudaMemcpyDeviceToHost);
        cudaMemcpy(sources_h->fine_flux_arr, sources_d->fine_flux_arr, N_fine * sizeof(float), cudaMemcpyDeviceToHost);
        cudaMemcpy(sources_h->sigT_arr, sources_d->sigT_arr, N_sigT * sizeof(float), cudaMemcpyDeviceToHost);

    } else {
        // Copy source data from host to GPU
        cudaMalloc((void **)&sources_d->fine_source_arr, N_fine * sizeof(float));
        cudaMalloc((void **)&sources_d->fine_flux_arr, N_fine * sizeof(float));
        cudaMalloc((void **)&sources_d->sigT_arr, N_sigT * sizeof(float));

        cudaMemcpy(sources_d->fine_source_arr, sources_h->fine_source_arr, N_fine * sizeof(float), cudaMemcpyHostToDevice);
        cudaMemcpy(sources_d->fine_flux_arr, sources_h->fine_flux_arr, N_fine * sizeof(float), cudaMemcpyHostToDevice);
        cudaMemcpy(sources_d->sigT_arr, sources_h->sigT_arr, N_sigT * sizeof(float), cudaMemcpyHostToDevice);

        // Manage source arrays on GPU
        #pragma omp target map(inout: sources_d->fine_source_arr, sources_d->fine_flux_arr, sources_d->sigT_arr)
        {
            for (long i = 0; i < N_fine; i++) {
                sources_d->fine_source_arr[i] = exp(-i * dx);
                sources_d->fine_flux_arr[i] = 1 + (i * dx - 1) * sources_d->fine_source_arr[i];

            }
        }
    }

    if(sources_h != NULL) {
