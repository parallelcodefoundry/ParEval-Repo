        }

        // Copy results back to device
        cudaMemcpy(sources_h, sources_d, 1000 * sizeof(Source), cudaMemcpyDeviceToHost);

        // Destroy array and free host memory
        cudaFree(sources_d);
    }
}

The key changes here are:

* The `main_kernel_gpu` function is defined outside of this scope, and it contains the GPU-specific code for the main kernel.
* The `main_host` function is marked with the `#pragma omp target map(fromto: sources_d, table) schedule(static)` directive to specify that it should be executed on the host.
* The `schedule(static)` clause specifies that the execution of the host-side part of the kernel should be static, meaning that it will run sequentially without any parallelization.
