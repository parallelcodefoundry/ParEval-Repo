int main( int argc, char * argv[] )

	// Calculate Number of 3D Source Regions

        #pragma omp target map(from:table[0:1])

	#endif

	// Setup CUDA blocks / threads.  This section is now irrelevant for OpenMP offload.
	// The number of threads will be controlled by OpenMP.

        #pragma omp target map(tofrom: rng_states[0:I.streams])
	for (int i = 0; i < I.streams; i++) {
            rng_states[i] = i; //Initialize the RNG states.  Replace with actual initialization.
	}

	assert( I.segments >= N_flux_states );
        size_t flux_size = N_flux_states * I.egroups * sizeof(float);
        #pragma omp target enter data map(alloc: flux_states[0:N_flux_states*I.egroups])
        #pragma omp target map(tofrom: flux_states[0:N_flux_states*I.egroups])
	//init_flux_states<<< blocks, I.egroups >>> ( flux_states, N_flux_states, I, RNG_states );  No longer needed.
        for (int i = 0; i < N_flux_states * I.egroups; i++) {

        }

	center_print("SIMULATION", 79);
	border_print();

	//OpenMP timer variables
	double start_time, end_time;
	start_time = omp_get_wtime();
	
	// Setup kernel call block parameters.  This section is mostly irrelevant for OpenMP offload.
        int num_threads = omp_get_max_threads();


	// Run Simulation Kernel Loop.  Offload to device.
        #pragma omp target map(to: I, sources_d[0:I.source_3D_regions], SA_d, table_d, rng_states[0:I.streams], flux_states[0:N_flux_states*I.egroups]) map(from: flux_states[0:N_flux_states*I.egroups])
        #pragma omp parallel num_threads(num_threads)
	{
            int id = omp_get_thread_num();
            int blockId = id / I.egroups;
            int g = id % I.egroups;
            run_kernel_omp(I, sources_d, SA_d, table_d, rng_states, flux_states, N_flux_states, blockId, g);

	printf("Simulation Complete.\n");

	border_print();
	center_print("RESULTS SUMMARY", 79);
	border_print();

	double tpi = ((double) (time) /
			(double)I.segments / (double) I.egroups) * 1.0e9;
	printf("%-25s%.3f seconds\n", "Runtime:", time);
	printf("%-25s%.8lf ns\n", "Time per Intersection:", tpi);
	border_print();

        #pragma omp target exit data map(release: flux_states[0:N_flux_states*I.egroups]) map(release: rng_states[0:I.streams]) map(release: table_d)
	free(host_flux_states);
	return 0;
}
