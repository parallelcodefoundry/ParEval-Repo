```cmake
# CMakeLists.txt for SimpleMOC-kernel

# Minimum CMake version required
cmake_minimum_required(VERSION 3.10)

# Project name
project(SimpleMOC-kernel)

# Find Kokkos
find_package(Kokkos REQUIRED)

# Set C++ standard
set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Set compiler flags
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -pedantic")

# Add source files
add_executable(SimpleMOC-kernel
  kernel.cpp
  main.cpp
  init.cpp
  io.cpp
)

# Include directories
target_include_directories(SimpleMOC-kernel PUBLIC
  ${CMAKE_CURRENT_SOURCE_DIR}
)

# Link against Kokkos
target_link_libraries(SimpleMOC-kernel
  Kokkos::KokkosCore
)

# Enable CUDA support if requested
if (KOKKOS_BACKEND STREQUAL "CUDA")
  target_compile_definitions(SimpleMOC-kernel PUBLIC -DKOKKOS_BACKEND_CUDA)
  target_link_libraries(SimpleMOC-kernel
    Kokkos::KokkosCuda
  )
endif()

# Install target
install(TARGETS SimpleMOC-kernel
  DESTINATION bin
)
```
