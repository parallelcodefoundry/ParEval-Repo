```cmake
cmake_minimum_required(VERSION 3.10)
project(SimpleMOC-kernel)

# Find Kokkos
find_package(Kokkos REQUIRED)

# Find CUDA
find_package(CUDA REQUIRED)

# Set C++ standard
set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Set source files
set(SOURCE_FILES
  kernel.cpp
  main.cpp
  init.cpp
  io.cpp
)

# Set header files
set(HEADER_FILES
  SimpleMOC-kernel_header.hpp
)

# Add executable
add_executable(SimpleMOC-kernel ${SOURCE_FILES} ${HEADER_FILES})

# Link against Kokkos
target_link_libraries(SimpleMOC-kernel Kokkos::KokkosCore)

# Link against CUDA
target_link_libraries(SimpleMOC-kernel CUDA::CUDA)

# Set CUDA architecture
set(CUDA_ARCHITECTURE "52" CACHE STRING "CUDA architecture")
set_property(TARGET SimpleMOC-kernel PROPERTY CUDA_ARCHITECTURE ${CUDA_ARCHITECTURE})

# Set optimization flags
set(CMAKE_CXX_FLAGS_RELEASE "-O3 -march=native")
set(CMAKE_CXX_FLAGS_DEBUG "-g -O0")

# Set debug flags
option(DEBUG "Enable debug mode" OFF)
if(DEBUG)
  set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -g -O0")
endif()

# Set profiling flags
option(PROFILE "Enable profiling mode" OFF)
if(PROFILE)
  set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -pg")
endif()

# Set table flags
option(TABLE "Enable exponential table" OFF)
if(TABLE)
  add_definitions(-DTABLE)
endif()
```
