```cmake
cmake_minimum_required(VERSION 3.10)
project(SimpleMOC-kernel)

# Find the Kokkos library
find_package(Kokkos REQUIRED)

# Set the C++ standard
set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Add the source files
add_executable(SimpleMOC-kernel
  kernel.cpp
  main.cpp
  init.cpp
  io.cpp
)

# Link the Kokkos library
target_link_libraries(SimpleMOC-kernel ${Kokkos_LIBRARIES})

# Add include directories
target_include_directories(SimpleMOC-kernel PUBLIC
  ${CMAKE_CURRENT_SOURCE_DIR}
)

# Set the CUDA architecture if using the CUDA backend
if (KOKKOS_ENABLE_CUDA)
  set(CUDA_ARCHITECTURE "52" CACHE STRING "CUDA architecture")
  target_compile_definitions(SimpleMOC-kernel PUBLIC -DSM_VERSION=${CUDA_ARCHITECTURE})
endif()

# Add compiler flags
target_compile_options(SimpleMOC-kernel PUBLIC
  -Wall
  -Wextra
  -Wpedantic
)

# Add optimization flags if OPTIMIZE is ON
option(OPTIMIZE "Enable compiler optimizations" ON)
if (OPTIMIZE)
  target_compile_options(SimpleMOC-kernel PUBLIC -O3)
endif()

# Add debug flags if DEBUG is ON
option(DEBUG "Enable debug mode" OFF)
if (DEBUG)
  target_compile_options(SimpleMOC-kernel PUBLIC -g)
endif()

# Add profiling flags if PROFILE is ON
option(PROFILE "Enable profiling" OFF)
if (PROFILE)
  target_compile_options(SimpleMOC-kernel PUBLIC -pg)
endif()

# Add table flag if TABLE is ON
option(TABLE "Enable exponential table" OFF)
if (TABLE)
  target_compile_definitions(SimpleMOC-kernel PUBLIC -DTABLE)
endif()
```
