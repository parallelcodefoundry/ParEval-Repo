#ifndef XSBENCH_SHARED_HEADER_H
#define XSBENCH_SHARED_HEADER_H

// Header for shared utilities across XSBench versions

#include <stdio.h>
#include <stdlib.h>
#include <string.h> // Added for string manipulation functions
#include <stdint.h>
#include <omp.h> // Include OpenMP header


typedef struct{
        int nthreads;
        long n_isotopes;
        long n_gridpoints;
        long lookups; // Changed to long to handle larger values
        char * HM;
        int grid_type; // 0: Unionized Grid (default)    1: Nuclide Grid
        int hash_bins;
        long particles; // Changed to long to handle larger values
        int simulation_method;
        int binary_mode;
        int kernel_id;
        int num_iterations;
        int num_warmups;
        char *filename;
} Inputs;

typedef struct{
  double device_to_host_time;
  double kernel_time;
  double host_to_device_time;
} Profile;

// Function to print profiling results.  Handles both file and stdout output.
inline void print_profile(Profile profile, Inputs in) {
  FILE* output = in.filename ? fopen(in.filename, "w") : stdout; // Use stdout if filename is NULL

  if (output) {  // Check if file was successfully opened or stdout is used
    fprintf(output, "host_to_device_ms,kernel_ms,device_to_host_ms,num_iterations,num_warmups\n");
    fprintf(output, "%f,%f,%f,%d,%d\n",
            profile.host_to_device_time*1000,
            profile.kernel_time*1000,
            profile.device_to_host_time*1000,
            in.num_iterations,
            in.num_warmups);
    if (in.filename) fclose(output); // Close the file only if it was opened
  } else {
    fprintf(stderr, "Error opening file for output.\n"); // Error message if file opening failed
  }
}


#endif // XSBENCH_SHARED_HEADER_H
