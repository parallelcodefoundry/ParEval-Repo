```makefile
#===============================================================================
# User Options
#===============================================================================

COMPILER    ?= clang++
OPTIMIZE    ?= yes
DEBUG       ?= no
PROFILE     ?= no

#===============================================================================
# Program name & source code list
#===============================================================================

program = XSBench

source = \
Main.cpp \
io.cpp \
Simulation.cpp \
GridInit.cpp \
XSutils.cpp \
Materials.cpp

obj = $(source:.cpp=.o)

#===============================================================================
# Sets Flags
#===============================================================================

# Standard Flags
CFLAGS := -std=c++17 -I/path/to/Kokkos

# Linker Flags
LDFLAGS = -lm -L/path/to/Kokkos -lkokkos_core

# Compiler
CC = $(COMPILER)

# Debug Flags
ifeq ($(DEBUG),yes)
  CFLAGS += -g -O0
  LDFLAGS  += -g -O0
endif

# Profiling Flags
ifeq ($(PROFILE),yes)
  CFLAGS += -pg
  LDFLAGS  += -pg
endif

# Optimization Flags
ifeq ($(OPTIMIZE),yes)
  CFLAGS += -O3
endif

#===============================================================================
# Targets to Build
#===============================================================================

$(program): $(obj) XSbench_header.hpp Makefile
	$(CC) $(CFLAGS) $(obj) -o $@ $(LDFLAGS)

%.o: %.cpp XSbench_header.hpp Makefile
	$(CC) $(CFLAGS) -c $< -o $@

clean:
	rm -rf $(program) $(obj)

edit:
	vim -p $(source) XSbench_header.hpp

run:
	./$(program)
```

**Note:**  This Makefile assumes Kokkos is installed and that `/path/to/Kokkos` is replaced with the actual Kokkos installation directory. You'll need to adjust this path accordingly.  The `-lkokkos_core` flag links the Kokkos core library. You might need additional libraries depending on your Kokkos configuration (e.g., for MPI support if you enable it in your Kokkos build).  Remember to adapt the compilation and linking options to your specific Kokkos setup and desired features (like MPI).  The CUDA specific flags (`-arch=sm_$(SM_VERSION)`) are removed as they are not needed for Kokkos.
