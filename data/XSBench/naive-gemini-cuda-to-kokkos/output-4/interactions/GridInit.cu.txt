#include "XSbench_header.hpp"

// Moves all required data structures to the Kokkos's memory space
SimulationData move_simulation_data_to_device(Inputs in, int mype, SimulationData SD) {
    if (mype == 0) printf("Allocating and moving simulation data to Kokkos memory space...\n");

    ////////////////////////////////////////////////////////////////////////////////
    // SUMMARY: Simulation Data Structure Manifest for "SD" Object
    // Here we list all heap arrays (and lengths) in SD that would need to be
    // offloaded manually if using an accelerator with a separate memory space
    ////////////////////////////////////////////////////////////////////////////////
    // int * num_nucs;                     // Length = length_num_nucs;
    // double * concs;                     // Length = length_concs
    // int * mats;                         // Length = length_mats
    // double * unionized_energy_array;    // Length = length_unionized_energy_array
    // int * index_grid;                   // Length = length_index_grid
    // NuclideGridPoint * nuclide_grid;    // Length = length_nuclide_grid
    //
    // Note: "unionized_energy_array" and "index_grid" can be of zero length
    //        depending on lookup method.
    //
    // Note: "Lengths" are given as the number of objects in the array, not the
    //       number of bytes.
    ////////////////////////////////////////////////////////////////////////////////

    //Kokkos View declarations.  The Kokkos::LayoutLeft is used to match the original code's array ordering.

    Kokkos::View<int*, Kokkos::HostSpace>  h_num_nucs("h_num_nucs", SD.length_num_nucs);
    Kokkos::View<double*, Kokkos::HostSpace> h_concs("h_concs", SD.length_concs);
    Kokkos::View<int*, Kokkos::HostSpace> h_mats("h_mats", SD.length_mats);
    Kokkos::View<double*, Kokkos::HostSpace> h_unionized_energy_array("h_unionized_energy_array", SD.length_unionized_energy_array);
    Kokkos::View<int*, Kokkos::HostSpace> h_index_grid("h_index_grid", SD.length_index_grid);
    Kokkos::View<NuclideGridPoint*, Kokkos::HostSpace> h_nuclide_grid("h_nuclide_grid", SD.length_nuclide_grid);
    Kokkos::View<unsigned long*, Kokkos::HostSpace> h_verification("h_verification", in.lookups);


    // Copy data from host to Kokkos Views
    Kokkos::deep_copy(h_num_nucs, SD.num_nucs);
    Kokkos::deep_copy(h_concs, SD.concs);
    Kokkos::deep_copy(h_mats, SD.mats);
    Kokkos::deep_copy(h_unionized_energy_array, SD.unionized_energy_array);
    Kokkos::deep_copy(h_index_grid, SD.index_grid);
    Kokkos::deep_copy(h_nuclide_grid, SD.nuclide_grid);


    //Create Kokkos Views in the execution space.  'execution_space' should be replaced by the desired Kokkos execution space (e.g., Kokkos::Cuda).  
    Kokkos::View<int*, Kokkos::DeviceSpace> d_num_nucs("d_num_nucs", SD.length_num_nucs);
    Kokkos::View<double*, Kokkos::DeviceSpace> d_concs("d_concs", SD.length_concs);
    Kokkos::View<int*, Kokkos::DeviceSpace> d_mats("d_mats", SD.length_mats);
    Kokkos::View<double*, Kokkos::DeviceSpace> d_unionized_energy_array("d_unionized_energy_array", SD.length_unionized_energy_array);
    Kokkos::View<int*, Kokkos::DeviceSpace> d_index_grid("d_index_grid", SD.length_index_grid);
    Kokkos::View<NuclideGridPoint*, Kokkos::DeviceSpace> d_nuclide_grid("d_nuclide_grid", SD.length_nuclide_grid);
    Kokkos::View<unsigned long*, Kokkos::DeviceSpace> d_verification("d_verification", in.lookups);

    //Copy data from HostSpace Views to DeviceSpace Views
    Kokkos::deep_copy(d_num_nucs, h_num_nucs);
    Kokkos::deep_copy(d_concs, h_concs);
    Kokkos::deep_copy(d_mats, h_mats);
    Kokkos::deep_copy(d_unionized_energy_array, h_unionized_energy_array);
    Kokkos::deep_copy(d_index_grid, h_index_grid);
    Kokkos::deep_copy(d_nuclide_grid, h_nuclide_grid);
    Kokkos::deep_copy(d_verification, h_verification);

    // Shallow copy of CPU simulation data to GPU simulation data.  We only copy pointers as the data is already moved to device memory.
    SimulationData GSD = SD;
    GSD.num_nucs = d_num_nucs.data();
    GSD.concs = d_concs.data();
    GSD.mats = d_mats.data();
    GSD.unionized_energy_array = d_unionized_energy_array.data();
    GSD.index_grid = d_index_grid.data();
    GSD.nuclide_grid = d_nuclide_grid.data();
    GSD.verification = d_verification.data();


    if (mype == 0) printf("GPU Initialization complete.\n");

    return GSD;
}

// Release device memory
void release_device_memory(SimulationData GSD) {
    //Kokkos Views are automatically deallocated when they go out of scope.  No explicit deallocation needed.
}

void release_memory(SimulationData SD) {
    //Free host-side memory
    free(SD.num_nucs);
    free(SD.concs);
    free(SD.mats);
    if (SD.length_unionized_energy_array > 0) free(SD.unionized_energy_array);
    free(SD.nuclide_grid);
    free(SD.verification);
}
