```cmake
cmake_minimum_required(VERSION 3.21)

project(
    XSBench_Kokkos
    VERSION 1.0
    LANGUAGES CXX
)

if(NOT CMAKE_BUILD_TYPE)
  set(CMAKE_BUILD_TYPE Release)
endif()

set(CMAKE_CXX_FLAGS "-Wall -Wextra")
set(CMAKE_CXX_FLAGS_DEBUG "-g")
set(CMAKE_CXX_FLAGS_RELEASE "-O3")

# Find Kokkos
find_package(Kokkos REQUIRED)

# Kokkos-specific compile flags.  These are appended to CMAKE_CXX_FLAGS.
set(KOKKOS_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${Kokkos_CXX_FLAGS}")

set(SOURCE Main.cpp io.cpp Simulation.cpp GridInit.cpp XSutils.cpp Materials.cpp)
add_executable(XSBench ${SOURCE} XSbench_header.hpp XSbench_shared_header.h)

# Link Kokkos libraries
target_link_libraries(XSBench Kokkos::Kokkos)


install(TARGETS XSBench DESTINATION bin)
```
