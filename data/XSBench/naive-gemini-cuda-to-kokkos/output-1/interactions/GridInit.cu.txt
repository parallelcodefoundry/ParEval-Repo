// Release Kokkos device memory
void release_device_memory(SimulationData KSD) {
    KSD.num_nucs.reset();
    KSD.concs.reset();
    KSD.mats.reset();
    if (KSD.length_unionized_energy_array > 0) KSD.unionized_energy_array.reset();
    KSD.nuclide_grid.reset();
    KSD.verification.reset();
}
