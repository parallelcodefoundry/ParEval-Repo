//Helper function for printing profiles to file
inline void print_profile(const Profile& profile, const Inputs& in) {
  std::string filename = in.filename ? in.filename : ""; //Handle defaulting to stdout.

  std::ofstream outputFile;
  if (filename.empty()) {
    outputFile.rdbuf()->pubsetbuf(0, 0); //Set to unbuffered for stdout
  }
  else{
    outputFile.open(filename);
  }

  outputFile << "host_to_device_ms,kernel_ms,device_to_host_ms,num_iterations,num_warmups" << std::endl;
  outputFile << profile.host_to_device_time * 1000 << ","
             << profile.kernel_time * 1000 << ","
             << profile.device_to_host_time * 1000 << ","
             << in.num_iterations << ","
             << in.num_warmups << std::endl;
  outputFile.close();
}
