Here is the translated CMakeLists.txt file:

```
cmake_minimum_required(VERSION 3.21)

project(
    XSBench
    VERSION 1.0
    LANGUAGES CXX OPENMP
)

if(NOT CMAKE_BUILD_TYPE)
  set(CMAKE_BUILD_TYPE Release)
endif()

set(CMAKE_CXX_FLAGS "-Wall -Wextra")
set(CMAKE_CXX_FLAGS_DEBUG "-g")
set(CMAKE_CXX_FLAGS_RELEASE "-O3")

set(SOURCE Main.c io.c Simulation.c GridInit.c XSutils.c Materials.c)

set_source_files_properties(${SOURCE} PROPERTIES LANGUAGE C)

add_executable(XSBench ${SOURCE})

install(TARGETS XSBench DESTINATION bin)
```

Note that we have added `OPENMP` to the `LANGUAGES` list and changed `CXX` to `C` for the source file properties. This will allow OpenMP offloading to be enabled in the build process.

Also, make sure to enable OpenMP offloading by adding `-fopenmp-targets=nvptx64-nvidia-cuda` to the C++ flags:

```
set(CMAKE_CXX_FLAGS "-Wall -Wextra -fopenmp-targets=nvptx64-nvidia-cuda")
```
