#include "XSbench_header.h"

// Moves all required data structures to the GPU's memory space (modified to use OpenMP and offload)
SimulationData move_simulation_data_to_device( Inputs in, int mype, SimulationData SD )
{
    #pragma omp target teams distribute parallel for
    if(mype == 0) printf("Allocating and moving simulation data to GPU memory space...\n");

    // Keep track of how much data we're allocating
    size_t nbytes = 0;

    // Shallow copy of CPU simulation data to GPU simulation data
    SimulationData GSD = SD;

    // Move data to GPU memory space
    #pragma omp target teams distribute parallel for
    for(int i = 0; i < in.n_isotopes; i++) {
        size_t sz = sizeof(NuclideGridPoint);
        gpuErrchk( cudaMalloc((void **) &GSD.nuclide_grid[i * in.n_gridpoints], sz) );
        gpuErrchk( cudaMemcpy(GSD.nuclide_grid + i * in.n_gridpoints, SD.nuclide_grid + i * in.n_gridpoints, sz, cudaMemcpyHostToDevice) );
        nbytes += sz;
    }

    #pragma omp target teams distribute parallel for
    if (in.grid_type == UNIONIZED || in.grid_type == HASH)
    {
        size_t sz = sizeof(double);
        gpuErrchk( cudaMalloc((void **) &GSD.unionized_energy_array, sz * in.n_isotopes * in.n_gridpoints) );
        gpuErrchk( cudaMemcpy(GSD.unionized_energy_array, SD.unionized_energy_array, sz * in.n_isotopes * in.n_gridpoints, cudaMemcpyHostToDevice) );
        nbytes += sz * in.n_isotopes * in.n_gridpoints;
    }

    if (in.grid_type == UNIONIZED)
    {
        size_t sz = sizeof(int);
        gpuErrchk( cudaMalloc((void **) &GSD.index_grid, sz * in.n_isotopes * in.n_gridpoints) );
        gpuErrchk( cudaMemcpy(GSD.index_grid, SD.index_grid, sz * in.n_isotopes * in.n_gridpoints, cudaMemcpyHostToDevice) );
        nbytes += sz * in.n_isotopes * in.n_gridpoints;
    }

    // Allocate verification array on device. This structure is not needed on CPU, so we don't
    // have to copy anything over.
    size_t sz = in.lookups * sizeof(unsigned long);
    gpuErrchk( cudaMalloc((void **) &GSD.verification, sz) );
    nbytes += sz;
    GSD.length_verification = in.lookups;

    // Synchronize
    #pragma omp target teams distribute parallel for
    gpuErrchk( cudaPeekAtLastError() );
    gpuErrchk( cudaDeviceSynchronize() );

    if(mype == 0 ) printf("GPU Intialization complete. Allocated %.0lf MB of data on GPU.\n", nbytes/1024.0/1024.0 );

    return GSD;
}

This translated code uses OpenMP's `teams distribute parallel for` directive to offload the computation to the GPU, and assumes that `cudaMalloc` and `cudaMemcpy` are available for memory management on the GPU. The `gpuErrchk` function is also assumed to be a custom error checking function specific to this project.
