```make
#===============================================================================
# User Options
#===============================================================================

COMPILER    ?= clang
CUDA_ARCH   ?= 80
OPTIMIZE    ?= yes
DEBUG       ?= no
PROFILE     ?= no
MPI         ?= no

#===============================================================================
# Program name & source code list
#===============================================================================

program = XSBench

source = \
Main.cu \
io.cu \
Simulation.cu \
GridInit.cu \
XSutils.cu \
Materials.cu

obj = $(source:.cu=.o)

#===============================================================================
# Sets Flags
#===============================================================================

# Standard Flags
CFLAGS := -std=c++14

# OpenMP Flags
CFLAGS += -fopenmp -fopenmp-targets=nvptx64-nvidia-cuda=10.0

# Linker Flags
LDFLAGS = -lm -lopenomp

# Compiler
ifeq ($(COMPILER),clang)
  CC = clang++
endif
ifeq ($(COMPILER),gcc)
  CC = g++
endif
ifeq ($(COMPILER),nvcc)
  CC = nvcc
endif

#===============================================================================
# Targets to Build
#===============================================================================

$(program): $(obj) XSbench_header.cuh Makefile
	$(CC) -O3 $(CFLAGS) $(obj) -o $@ $(LDFLAGS)

%.o: %.cu XSbench_header.cuh Makefile
	$(CC) -c $(CFLAGS) -fopenmp-targets=nvptx64-nvidia-cuda=10.0 $< -o $@

clean:
	rm -rf $(program) $(obj)

edit:
	vim -p $(source) XSbench_header.cuh

run:
	./$(program)
```

Note: The `.cu` files will be compiled with OpenMP enabled, but since they are CUDA sources, the `-fopenmp-targets=nvptx64-nvidia-cuda=10.0` flag is used to enable offloading of OpenMP kernels to a CUDA device.

The `cuda-arch` and `compiler` variables are assumed to be set by the user when running `make`. The rest of the Makefile should work similarly as before, but with OpenMP enabled and CUDA architecture specified.
