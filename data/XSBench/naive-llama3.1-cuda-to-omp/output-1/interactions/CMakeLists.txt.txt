Here is the translated `CMakeLists.txt` file for the OpenMP-offload execution model:

```cmake
cmake_minimum_required(VERSION 3.21)

project(
    XSBench_OpenMP
    VERSION 1.0
    LANGUAGES CXX
)

if(NOT CMAKE_BUILD_TYPE)
  set(CMAKE_BUILD_TYPE Release)
endif()

set(CMAKE_CXX_FLAGS "-Wall -Wextra")
set(CMAKE_CXX_FLAGS_DEBUG "-g")
set(CMAKE_CXX_FLAGS_RELEASE "-O3")

set(SOURCE Main.c io.c Simulation.c GridInit.c XSutils.c Materials.c)

set_source_files_properties(${SOURCE} PROPERTIES LANGUAGE CXX)

add_executable(XSBench ${SOURCE})

target_link_libraries(XSBench -fopenmp)

set(CMAKE_OFFLOAD_ARCHITECTURES "cuda")
set_target_offload_properties(XSBench
        CUDA_SOURCE_FILES Main.cu io.cu Simulation.cu GridInit.cu XSutils.cu Materials.cu)
```
