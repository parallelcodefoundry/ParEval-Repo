Here's how you can translate the `XSbench_shared_header.h` file from CUDA to OpenMP offloading:

```c
#ifndef XSBENCH_SHARED_HEADER_H
#define XSBENCH_SHARED_HEADER_H

// Header for shared utilities across XSBench versions

typedef struct{
        int nthreads;
        long n_isotopes;
        long n_gridpoints;
        int lookups;
        char * HM;
        int grid_type; // 0: Unionized Grid (default)    1: Nuclide Grid
        int hash_bins;
        int particles;
        int simulation_method;
        int binary_mode;
        int kernel_id;
        int num_iterations;
        int num_warmups;
        char *filename;
} Inputs;

typedef struct{
  double device_to_host_time;
  double kernel_time;
  double host_to_device_time;
} Profile;

#pragma omp declare target

inline void print_profile(Profile profile, Inputs in) {
  #pragma omp target enter data map(to:profile) map(from:in.filename)
  if (in.filename) {
    FILE* output = fopen(in.filename, "w");
    fprintf(output, "host_to_device_ms,kernel_ms,device_to_host_ms,num_iterations,num_warmups\n");
    fprintf(output, "%f,%f,%f,%d,%d\n",
            profile.host_to_device_time*1000,
            profile.kernel_time*1000,
            profile.device_to_host_time*1000,
            in.num_iterations,
            in.num_warmups);
    fclose(output);
  }
  else {
    #pragma omp target data map(from:in)
    printf("host_to_device_ms,kernel_ms,device_to_host_ms,num_iterations,num_warmups\n");
    printf("%f,%f,%f,%d,%d\n",
           profile.host_to_device_time*1000,
           profile.kernel_time*1000,
           profile.device_to_host_time*1000,
           in.num_iterations,
           in.num_warmups);
  }
}

#pragma omp declare target

#endif // XSBENCH_SHARED_HEADER_H
```

Please note that the OpenMP offloading model will create a device region where data is mapped, and in this case, we are mapping `profile` to the host side and `in.filename` to the host. The `print_profile` function has been modified to use OpenMP directives for target entry data and target data.

The rest of the file remains unchanged as it does not contain any CUDA specific code that would need translation. 

Also note that you may need to modify your Makefile to correctly link against the OpenMP library. This can be done by adding `-fopenmp` to your compiler flags in the `CFLAGS` section.

For example, if your Makefile looks like this:

```make
# Sets Flags
CFLAGS := 

...

ifeq ($(COMPILER),nvidia)
  CC = nvcc
  CFLAGS += -Xcompiler -Wall -Xcompiler -O3 -arch=sm_$(SM_VERSION) -std=c++14
endif

# ... other compiler settings ...
```

You can add `-fopenmp` to the `CFLAGS` like this:

```make
ifeq ($(COMPILER),nvidia)
  CC = nvcc
  CFLAGS += -Xcompiler -Wall -Xcompiler -O3 -arch=sm_$(SM_VERSION) -std=c++14 -fopenmp
endif

# ... other compiler settings ...
```

Please note that the `-fopenmp` flag is added only for the Nvidia compiler, as it's specific to OpenMP offloading on accelerators.
