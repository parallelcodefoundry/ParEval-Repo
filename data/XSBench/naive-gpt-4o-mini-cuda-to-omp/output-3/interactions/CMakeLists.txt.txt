Here is the translated `CMakeLists.txt` file for the OpenMP Offload execution model:

```cmake
cmake_minimum_required(VERSION 3.21)

project(
    XSBench_OpenMP
    VERSION 1.0
    LANGUAGES CXX
)

if(NOT CMAKE_BUILD_TYPE)
  set(CMAKE_BUILD_TYPE Release)
endif()

set(CMAKE_CXX_FLAGS "-Wall -Wextra")
set(CMAKE_CXX_FLAGS_DEBUG "-g")
set(CMAKE_CXX_FLAGS_RELEASE "-O3")

set(SOURCE Main.cu io.cu Simulation.cu GridInit.cu XSutils.cu Materials.cu)

# Specify OpenMP as the parallelization model
find_package(OpenMP REQUIRED)

if(OpenMP_CXX_FOUND)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${OpenMP_CXX_FLAGS}")
endif()

add_executable(XSBench ${SOURCE})

install(TARGETS XSBench DESTINATION bin)
```

### Changes Made:
1. The project name has been changed to `XSBench_OpenMP` to reflect the new execution model.
2. The CUDA language specification has been removed since we are transitioning to OpenMP.
3. The `find_package(OpenMP REQUIRED)` line has been added to locate the OpenMP package.
4. The OpenMP flags are appended to the `CMAKE_CXX_FLAGS` if OpenMP is found. 

This configuration will allow you to compile the project using OpenMP for parallel execution.
