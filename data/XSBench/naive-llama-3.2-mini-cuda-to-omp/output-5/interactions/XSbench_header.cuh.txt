```c++
#ifndef XSBECH_OFFLOAD_SHARED_HEADER_H
#define XSBECH_OFFLOAD_SHARED_HEADER_H

// Header for shared utilities across XSBench versions

typedef struct{
        int nthreads;
        long n_isotopes;
        long n_gridpoints;
        int lookups;
        char * HM;
        int grid_type; // 0: Unionized Grid (default)    1: Nuclide Grid
        int hash_bins;
        int particles;
        int simulation_method;
        int binary_mode;
        int kernel_id;
        int num_iterations;
        int num_warmups;
        char *filename;
} Inputs;

typedef struct{
  double device_to_host_time;
  double kernel_time;
  double host_to_device_time;
} Profile;

inline void print_profile(Profile profile, Inputs in) {
#ifdef _OPENMP
#pragma omp parallel for
#endif
    if (in.filename) {
        FILE* output = fopen(in.filename, "w");
        fprintf(output, "host_to_device_ms,kernel_ms,device_to_host_ms,num_iterations,num_warmups\n");
        fprintf(output, "%f,%f,%f,%d,%d\n",
                profile.host_to_device_time*1000,
                profile.kernel_time*1000,
                profile.device_to_host_time*1000,
                in.num_iterations,
                in.num_warmups);
        fclose(output);
    }
    else {
#ifdef _OPENMP
#pragma omp parallel for
#endif
        printf("host_to_device_ms,kernel_ms,device_to_host_ms,num_iterations,num_warmups\n");
        printf("%f,%f,%f,%d,%d\n",
               profile.host_to_device_time*1000,
               profile.kernel_time*1000,
               profile.device_to_host_time*1000,
               in.num_iterations,
               in.num_warmups);
    }
}

#endif // XSBECH_OFFLOAD_SHARED_HEADER_H
```
Note that I added the `#ifdef _OPENMP` and corresponding pragmas to enable parallelization only when compiled with OpenMP. This allows for offloading execution to a separate thread or process, which can improve performance on multi-core systems.
