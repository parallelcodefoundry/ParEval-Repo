```c
#ifndef XSBENCH_SHARED_HEADER_H_OPENMP_OFFLOAD
#define XSBENCH_SHARED_HEADER_H_OPENMP_OFFLOAD

#include <omp.h>

typedef struct{
        int nthreads;
        long n_isotopes;
        long n_gridpoints;
        int lookups;
        char * HM;
        int grid_type; // 0: Unionized Grid (default)    1: Nuclide Grid
        int hash_bins;
        int particles;
        int simulation_method;
        int binary_mode;
        int kernel_id;
        int num_iterations;
        int num_warmups;
        char *filename;
} Inputs;

typedef struct{
  double device_to_host_time;
  double kernel_time;
  double host_to_device_time;
} Profile;

inline void print_profile(Profile profile, Inputs in) {
#pragma omp parallel for
    for(int i = 0; i < in.num_iterations; ++i) {
        if (in.filename) {
            FILE* output = fopen(in.filename, "w");
            fprintf(output, "host_to_device_ms,kernel_ms,device_to_host_ms,num_iterations,num_warmups\n");
            fprintf(output, "%f,%f,%f,%d,%d\n",
                    profile.host_to_device_time*1000,
                    profile.kernel_time*1000,
                    profile.device_to_host_time*1000,
                    i,
                    in.num_warmups);
            fclose(output);
        }
        else {
            printf("host_to_device_ms,kernel_ms,device_to_host_ms,num_iterations,num_warmups\n");
            printf("%f,%f,%f,%d,%d\n",
                   profile.host_to_device_time*1000,
                   profile.kernel_time*1000,
                   profile.device_to_host_time*1000,
                   i,
                   in.num_warmups);
        }
    }
}

#endif // XSBENCH_SHARED_HEADER_H_OPENMP_OFFLOAD
```

Note that I have used `#pragma omp parallel for` to execute the loop in parallel. This will be executed by the OpenMP runtime, which will divide the work among the threads specified by `omp_get_max_threads()`. The exact behavior may vary depending on the specific configuration of the compiler and the system being used.

Also note that I have removed the `#endif // XSBENCH_SHARED_HEADER_H` at the end of the file, since this is now an OpenMP-offload version of the header file. If you want to include additional code or definitions in the header file, you would need to add them before the `#pragma omp parallel for` directive.
