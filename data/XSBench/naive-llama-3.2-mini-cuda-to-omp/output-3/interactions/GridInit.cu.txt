#include <omp.h>
#include "GridInit.cu"

// Function to create grid dimensions and initialize it on device
__global__
void init_grid(int *d_grid, int n, int m) {
  #pragma omp target map(tofrom(d_grid))
  __shared__ float shared_cache[4096];
  
  // Initialize the first dimension of the grid
  d_grid[0] = omp_get_thread_num();
  
  // Launch a thread for each element in the grid on each SM
  #pragma omp parallel for num_threads(n*m)
  for (int i=1; i<n; i++) {
    d_grid[i*4] = i;
    
    // Launch a loop to initialize all elements of the current row
    __syncthreads();
    for (int j=0; j<m; j+=32) {
      #pragma omp target map(from(shared_cache))
      shared_cache[j] = 1.0f;
      
      // Initialize all elements in the current row with a non-zero value
      if (j < m) d_grid[i*4+j] = 0.5f;
    }
    
    __syncthreads();
    for (int j=0; j<32; j++) {
      #pragma omp target map(to(shared_cache))
      shared_cache[j] = 1.0f;
      
      // Initialize all elements of the current row with a non-zero value
      if (j < m) d_grid[i*4+j+32] = 1.0f;
    }
    
    __syncthreads();
    for (int j=0; j<64; j++) {
      #pragma omp target map(to(shared_cache))
      shared_cache[j] = 2.0f;
      
      // Initialize all elements of the current row with a non-zero value
      if (j < m) d_grid[i*4+j+64] = 3.0f;
    }
    
    __syncthreads();
    for (int j=0; j<128; j++) {
      #pragma omp target map(to(shared_cache))
      shared_cache[j] = 1.0f;
      
      // Initialize all elements of the current row with a non-zero value
      if (j < m) d_grid[i*4+j+128] = 5.0f;
    }
    
    __syncthreads();
    for (int j=0; j<256; j++) {
      #pragma omp target map(to(shared_cache))
      shared_cache[j] = 2.0f;
      
      // Initialize all elements of the current row with a non-zero value
      if (j < m) d_grid[i*4+j+256] = 6.0f;
    }
    
    __syncthreads();
    for (int j=0; j<512; j++) {
      #pragma omp target map(to(shared_cache))
      shared_cache[j] = 3.0f;
      
      // Initialize all elements of the current row with a non-zero value
      if (j < m) d_grid[i*4+j+512] = 7.0f;
    }
    
    __syncthreads();
    for (int j=0; j<1024; j++) {
      #pragma omp target map(to(shared_cache))
      shared_cache[j] = 4.0f;
      
      // Initialize all elements of the current row with a non-zero value
      if (j < m) d_grid[i*4+j+1024] = 8.0f;
    }
    
    __syncthreads();
    for (int j=0; j<2048; j++) {
      #pragma omp target map(to(shared_cache))
      shared_cache[j] = 5.0f;
      
      // Initialize all elements of the current row with a non-zero value
      if (j < m) d_grid[i*4+j+2048] = 9.0f;
    }
    
    __syncthreads();
    for (int j=0; j<4096; j++) {
      #pragma omp target map(to(shared_cache))
      shared_cache[j] = 6.0f;
      
      // Initialize all elements of the current row with a non-zero value
      if (j < m) d_grid[i*4+j+4096] = 10.0f;
    }
  }
}
