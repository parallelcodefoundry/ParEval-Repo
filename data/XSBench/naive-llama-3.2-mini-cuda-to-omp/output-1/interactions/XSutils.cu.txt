// XSMC.cu: Cuda kernel for XSBench Monte Carlo Reactor Analysis program

#include <stdio.h>

#define N 1000000 // Number of particles

void reduce_sum(double* a, double* sum) {
  __shared__ double temp[1024];

  int idx = blockIdx.x * blockDim.x + threadIdx.x;
  if (idx < a->n_gridpoints) {
    temp[threadIdx.x] = a[idx];
    __syncthreads();

    for (int i = threadIdx.x; i < a->n_gridpoints; i += blockDim.x) {
      temp[i] += a[i + blockDim.x * idx];
    }

    for (int i = blockDim.x; i > 0; i >>= 1)
      if (threadIdx.x < i) {
        temp[threadIdx.x] += temp[threadIdx.x + i];
      }
    __syncthreads();

    if (idx == 0 && a->n_gridpoints % blockDim.x == 0) {
      sum[0] = temp[threadIdx.x];
    } else if (threadIdx.x < a->n_gridpoints % blockDim.x) {
      sum[idx / blockDim.x + 1] += temp[threadIdx.x];
    }
  }
}
