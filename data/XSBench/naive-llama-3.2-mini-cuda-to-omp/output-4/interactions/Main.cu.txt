#include <stdio.h>
#include <omp.h>

#define EVENT_BASE 0
#define PARTICLE_BASE 1

void kernel_event(int* event, int* particle, int num_isotopes) {
    #pragma omp parallel for reduction(max:*) private(part)
    for (int i = 0; i < num_isotopes; ++i) {
        int idx = EVENT_BASE + i;
        int particle_idx = PARTICLE_BASE + i;

        // Binary search
        int low = 0;
        int high = num_isotopes - 1;
        while (low <= high) {
            int mid = (low + high) / 2;
            if (particle[mid] < idx) {
                low = mid + 1;
            } else {
                high = mid - 1;
            }
        }

        // Particle is in the correct iteration range
        if (high >= 0 && particle[high] == idx) {
            #pragma omp offload map(tofrom:particle[high]) schedule(inline, 1)
            particle[high]++;
        } else {
            // Particle is not in the correct iteration range
            return;
        }
    }
}
