// XSBench Shared Header for OpenMP-Offload Execution Model

#ifndef XSBENCH_SHARED_HEADER_H
#define XSBENCH_SHARED_HEADER_H

#include <omp.h>

// Header for shared utilities across XSBench versions

typedef struct{
    int nthreads;
    long n_isotopes;
    long n_gridpoints;
    int lookups;
    char * HM;
    int grid_type; // 0: Unionized Grid (default)    1: Nuclide Grid
    int hash_bins;
    int particles;
    int simulation_method;
    int binary_mode;
    int kernel_id;
    int num_iterations;
    int num_warmups;
    char *filename;
} Inputs;

// Function to print the profiling data for OpenMP-offload execution model
inline void print_profile(Profile profile, Inputs in) {
    if (in.filename) {
        FILE* output = fopen(in.filename, "w");
        fprintf(output, "host_to_device_ms,kernel_ms,device_to_host_ms,num_iterations,num_warmups\n");
        fprintf(output, "%f,%f,%f,%d,%d\n",
                profile.host_to_device_time*1000,
                profile.kernel_time*1000,
                profile.device_to_host_time*1000,
                in.num_iterations,
                in.num_warmups);
        fclose(output);
    }
    else {
        printf("host_to_device_ms,kernel_ms,device_to_host_ms,num_iterations,num_warmups\n");
        printf("%f,%f,%f,%d,%d\n",
               profile.host_to_device_time*1000,
               profile.kernel_time*1000,
               profile.device_to_host_time*1000,
               in.num_iterations,
               in.num_warmups);
    }
}

// Function to check if a particle can be simulated
inline bool can_simulate(Particle particle, Inputs in) {
    // Check if the particle is not yet warmed up and has enough particles for offloading
    if (particle.warmup_count == 0 && particle.num_particles > in.particles / in.nthreads) {
        return true;
    }
    else {
        return false;
    }
}

// Function to simulate a particle using OpenMP-offload execution model
inline void simulate_particle(Particle &particle, Inputs in) {
#ifdef OMP_OFFLOAD
#pragma omp offload target(device) map(tofrom)
    sim_kernel(particle);
#else
    // Simulation code here
#endif
}

// Function to perform the Monte Carlo simulation for a material
inline void simulate_material(Material *material, Inputs in) {
    for (int i = 0; i < material->n_particles; i++) {
        Particle particle;
        // Initialize the particle's properties
        particle = initialize_particle(material);
        if (can_simulate(particle, in)) {
            simulate_particle(particle, in);
        }
    }
}

// Function to perform the Monte Carlo simulation for all materials
inline void simulate_all_materials(Material *materials, Inputs in) {
    for (int i = 0; i < materials->n_materials; i++) {
        Material material;
        // Initialize the material's properties
        material = initialize_material(materials);
        simulate_material(&material, in);
    }
}

#endif // XSBENCH_SHARED_HEADER_H
