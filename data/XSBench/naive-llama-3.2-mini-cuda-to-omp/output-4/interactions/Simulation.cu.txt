__global__ void do_kernel(Inputs *in) {
    // Simulate the kernel execution here

void do_kernel_offload(Inputs *in, int num_particles) {
    #pragma omp target map(in: Inputs)
    __device__ void do_kernel_device(Inputs in) {
        // Simulate the kernel execution here
    }

    #pragma omp target map(to:do_kernel_device, from:in) 
    __global__ void do_kernel_offload_kernel() {
        int thread_id = blockIdx.x * blockDim.x + threadIdx.x;
        if (thread_id < num_particles) {
            do_kernel_device(in);
        }
    }

    cudaDeviceProp devProps;

    int block_size = 256;
    int grid_size = (num_particles + block_size - 1) / block_size;

    do_kernel_offload_kernel<<<grid_size, block_size>>>(in);
}

int main() {
    Inputs in;
    in.nparticles = MAX_PARTICLES; // Number of particles to simulate
    in.kernel_id = 0; // Kernel ID (0 for default kernel)

    int *host_particles = malloc(in.nparticles * sizeof(int));

    // Initialize the inputs

    // Generate random particle IDs
    for (int i = 0; i < in.nparticles; i++) {
        host_particles[i] = rand() % MAX_PARTICLES;
    }
