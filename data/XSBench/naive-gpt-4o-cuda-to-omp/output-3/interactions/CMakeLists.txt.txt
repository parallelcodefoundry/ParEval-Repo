```cmake
cmake_minimum_required(VERSION 3.21)

project(
    XSBench_OpenMP_Offload
    VERSION 1.0
    LANGUAGES C CXX
)

if(NOT CMAKE_BUILD_TYPE)
  set(CMAKE_BUILD_TYPE Release)
endif()

set(CMAKE_CXX_FLAGS "-Wall -Wextra")
set(CMAKE_CXX_FLAGS_DEBUG "-g")
set(CMAKE_CXX_FLAGS_RELEASE "-O3")

# Add OpenMP flags
find_package(OpenMP REQUIRED)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${OpenMP_CXX_FLAGS}")

set(SOURCE Main.cu io.cu Simulation.cu GridInit.cu XSutils.cu Materials.cu)

# Convert CUDA source files to C++ for OpenMP offloading
set_source_files_properties(${SOURCE} PROPERTIES LANGUAGE CXX)

add_executable(XSBench ${SOURCE})

# Link OpenMP
target_link_libraries(XSBench PRIVATE OpenMP::OpenMP_CXX)

install(TARGETS XSBench DESTINATION bin)
```

In this translation, the project name is updated to reflect the OpenMP offload model. The `LANGUAGES` field is set to `C CXX` since OpenMP offloading is typically used with C/C++ compilers. The `find_package(OpenMP REQUIRED)` command is used to locate the OpenMP package, and the necessary flags are added to the compilation process. The CUDA source files are treated as C++ files for OpenMP offloading, and OpenMP libraries are linked to the executable.
