Here is the translated `CMakeLists.txt` file for the openmp-offload execution model:

```cmake
cmake_minimum_required(VERSION 3.21)

project(
    XSBench_OpenMP
    VERSION 1.0
    LANGUAGES C CXX
)

if(NOT CMAKE_BUILD_TYPE)
  set(CMAKE_BUILD_TYPE Release)
endif()

set(CMAKE_C_FLAGS "-Wall -Wextra")
set(CMAKE_C_FLAGS_DEBUG "-g")
set(CMAKE_C_FLAGS_RELEASE "-O3")

# Add OpenMP flags
find_package(OpenMP REQUIRED)
set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} ${OpenMP_C_FLAGS}")

set(SOURCE Main.c io.c Simulation.c GridInit.c XSutils.c Materials.c)

add_executable(XSBench ${SOURCE})

target_link_libraries(XSBench PRIVATE OpenMP::OpenMP_C)

install(TARGETS XSBench DESTINATION bin)
```

### Key Changes:
- Changed the project name to `XSBench_OpenMP`.
- Changed the `LANGUAGES` to `C CXX` to reflect the use of C and C++.
- Added `find_package(OpenMP REQUIRED)` to locate the OpenMP package.
- Added `${OpenMP_C_FLAGS}` to the `CMAKE_C_FLAGS` to include OpenMP compilation flags.
- Linked the OpenMP library using `target_link_libraries(XSBench PRIVATE OpenMP::OpenMP_C)`.
