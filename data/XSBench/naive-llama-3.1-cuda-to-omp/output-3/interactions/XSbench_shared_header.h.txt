Here is the translated `XSbench_shared_header.h` file using OpenMP offloading:
```c
#ifndef XSBENCH_SHARED_HEADER_H
#define XSBENCH_SHARED_HEADER_H

// Header for shared utilities across XSBench versions

typedef struct{
        int nthreads;
        long n_isotopes;
        long n_gridpoints;
        int lookups;
        char * HM;
        int grid_type; // 0: Unionized Grid (default)    1: Nuclide Grid
        int hash_bins;
        int particles;
        int simulation_method;
        int binary_mode;
        int kernel_id;
        int num_iterations;
        int num_warmups;
        char *filename;
} Inputs;

typedef struct{
  double device_to_host_time;
  double kernel_time;
  double host_to_device_time;
} Profile;

#ifdef __OPENMP
#pragma omp declare target
#endif

inline void print_profile(Profile profile, Inputs in) {
  #ifdef __OPENMP
  #pragma omp target enter data map(to:profile, in)
  #endif
  
  if (in.filename) {
    FILE* output = fopen(in.filename, "w");
    fprintf(output, "host_to_device_ms,kernel_ms,device_to_host_ms,num_iterations,num_warmups\n");
    fprintf(output, "%f,%f,%f,%d,%d\n",
            profile.host_to_device_time*1000,
            profile.kernel_time*1000,
            profile.device_to_host_time*1000,
            in.num_iterations,
            in.num_warmups);
    fclose(output);
  }
  #ifdef __OPENMP
  #pragma omp target exit data map(delete:profile, in)
  #endif
  
  else {
    printf("host_to_device_ms,kernel_ms,device_to_host_ms,num_iterations,num_warmups\n");
    printf("%f,%f,%f,%d,%d\n",
           profile.host_to_device_time*1000,
           profile.kernel_time*1000,
           profile.device_to_host_time*1000,
           in.num_iterations,
           in.num_warmups);
  }
}

#ifdef __OPENMP
#pragma omp declare target
#endif

#endif // XSBENCH_SHARED_HEADER_H
```
Note that I've added OpenMP pragmas to offload the `print_profile` function and its arguments to a device (e.g., GPU). The `map(to:profile, in)` directive tells OpenMP to transfer the `profile` and `in` variables from host memory to device memory. The `exit data` directive is used to release the transferred data on exit.
