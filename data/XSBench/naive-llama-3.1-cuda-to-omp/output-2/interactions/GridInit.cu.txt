Here's the `GridInit.cu` file translated to the OpenMP offloading execution model:

```c
#include "XSbench_header.cuh"

// Moves all required data structures to the GPU's memory space
SimulationData move_simulation_data_to_device( Inputs in, int mype, SimulationData SD )
{
    if(mype == 0) printf("Allocating and moving simulation data to GPU memory space...\n");

    // Structure to hold all allocated simuluation data arrays
    SimulationData GSD;

    // Keep track of how much data we're allocating
    size_t nbytes = 0;

    #pragma omp target teams distribute parallel for map(to:GSD)
    {
        if(mype == 0) printf("Intializing nuclide grids...\n");

        // First, we need to initialize our nuclide grid. This comes in the form
        // of a flattened 2D array that hold all the information we need to define
        // the cross sections for all isotopes in the simulation.
        GSD.length_nuclide_grid = in.n_isotopes * in.n_gridpoints;
        GSD.nuclide_grid     = (NuclideGridPoint *) malloc( GSD.length_nuclide_grid * sizeof(NuclideGridPoint));
        assert(GSD.nuclide_grid != NULL);
        nbytes += GSD.length_nuclide_grid * sizeof(NuclideGridPoint);

        for( int i = 0; i < GSD.length_nuclide_grid; i++ )
        {
            GSD.nuclide_grid[i].energy        = LCG_random_double(&seed);
            GSD.nuclide_grid[i].total_xs      = LCG_random_double(&seed);
            GSD.nuclide_grid[i].elastic_xs    = LCG_random_double(&seed);
            GSD.nuclide_grid[i].absorbtion_xs = LCG_random_double(&seed);
            GSD.nuclide_grid[i].fission_xs    = LCG_random_double(&seed);
            GSD.nuclide_grid[i].nu_fission_xs = LCG_random_double(&seed);
        }

        // Sort so that each nuclide has data stored in ascending energy order.
        for( int i = 0; i < in.n_isotopes; i++ )
                qsort( &GSD.nuclide_grid[i*in.n_gridpoints], in.n_gridpoints, sizeof(NuclideGridPoint), NGP_compare);

        // error debug check
        /*
        for( int i = 0; i < in.n_isotopes; i++ )
        {
            printf("NUCLIDE %d ==============================\n", i);
            for( int j = 0; j < in.n_gridpoints; j++ )
                printf("E%d = %lf\n", j, GSD.nuclide_grid[i * in.n_gridpoints + j].energy);
        }
        */

        // Allocate Verification Array
        size_t sz = in.lookups * sizeof(unsigned long);
        GSD.verification = (unsigned long *) malloc(sz);
        nbytes += sz;
    }

    // Synchronize
    #pragma omp target teams distribute parallel for map(to:GSD)
    {
        if(mype == 0) printf("Intializing unionized grid...\n");

        // Allocate space to hold the union of all nuclide energy data
        GSD.length_unionized_energy_array = in.n_isotopes * in.n_gridpoints;
        GSD.unionized_energy_array = (double *) malloc( GSD.length_unionized_energy_array * sizeof(double));
        assert(GSD.unionized_energy_array != NULL );
        nbytes += GSD.length_unionized_energy_array * sizeof(double);

        // Copy energy data over from the nuclide energy grid
        for( int i = 0; i < GSD.length_unionized_energy_array; i++ )
                GSD.unionized_energy_array[i] = GSD.nuclide_grid[i].energy;

        // Sort unionized energy array
        qsort( GSD.unionized_energy_array, GSD.length_unionized_energy_array, sizeof(double), double_compare);

        // Allocate space to hold the acceleration grid indices
        GSD.length_index_grid = GSD.length_unionized_energy_array * in.n_isotopes;
        GSD.index_grid = (int *) malloc( GSD.length_index_grid * sizeof(int));
        assert(GSD.index_grid != NULL);
        nbytes += GSD.length_index_grid * sizeof(int);

        // Generates the double indexing grid
        int * idx_low = (int *) calloc( in.n_isotopes, sizeof(int));
        assert(idx_low != NULL );
        double * energy_high = (double *) malloc( in.n_isotopes * sizeof(double));
        assert(energy_high != NULL );

        for( int i = 0; i < in.n_isotopes; i++ )
                energy_high[i] = GSD.nuclide_grid[i * in.n_gridpoints + 1].energy;

        for( long e = 0; e < GSD.length_unionized_energy_array; e++ )
        {
            double unionized_energy = GSD.unionized_energy_array[e];
            for( long i = 0; i < in.n_isotopes; i++ )
            {
                if( unionized_energy < energy_high[i]  )
                    GSD.index_grid[e * in.n_isotopes + i] = idx_low[i];
                else if( idx_low[i] == in.n_gridpoints - 2 )
                    GSD.index_grid[e * in.n_isotopes + i] = idx_low[i];
                else
                {
                    idx_low[i]++;
                    GSD.index_grid[e * in.n_isotopes + i] = idx_low[i];
                    energy_high[i] = GSD.nuclide_grid[i * in.n_gridpoints + idx_low[i] + 1].energy;
                }
            }
        }

        free(idx_low);
        free(energy_high);
    }

    // Synchronize
    #pragma omp target teams distribute parallel for map(to:GSD)
    {
        if(mype == 0) printf("Intializing material data...\n");

        // Set the number of nuclides in each material
        GSD.num_nucs  = load_num_nucs(in.n_isotopes);
        GSD.length_num_nucs = 12; // There are always 12 materials in XSBench

        // Intialize the flattened 2D grid of material data. The grid holds
        // a list of nuclide indices for each of the 12 material types. The
        // grid is allocated as a full square grid, even though not all
        // materials have the same number of nuclides.
        GSD.mats = load_mats(GSD.num_nucs, in.n_isotopes, &GSD.max_num_nucs);
        GSD.length_mats = GSD.length_num_nucs * GSD.max_num_nucs;

        // Intialize the flattened 2D grid of nuclide concentration data. The grid holds
        // a list of nuclide concentrations for each of the 12 material types. The
        // grid is allocated as a full square grid, even though not all
        // materials have the same number of nuclides.
        GSD.concs = load_concs(GSD.num_nucs, GSD.max_num_nucs);
        GSD.length_concs = GSD.length_mats;
    }

    if(mype == 0) printf("Intialization complete. Allocated %.0lf MB of data on CPU.\n", nbytes/1024.0/1024.0 );

    return GSD;
}

// Release device memory
void release_device_memory(SimulationData GSD) {
    #pragma omp target teams distribute parallel for map(from:GSD)
    {
        cudaFree(GSD.num_nucs);
        cudaFree(GSD.concs);
        cudaFree(GSD.mats);
        if (GSD.length_unionized_energy_array > 0) cudaFree(GSD.unionized_energy_array);
        cudaFree(GSD.nuclide_grid);
        cudaFree(GSD.verification);
    }
}

void release_memory(SimulationData SD) {
    free(SD.num_nucs);
    free(SD.concs);
    free(SD.mats);
    if (SD.length_unionized_energy_array > 0) free(SD.unionized_energy_array);
    free(SD.nuclide_grid);
    free(SD.verification);
}
```

Note that the code has been modified to use OpenMP offloading instead of CUDA. The `#pragma omp target teams distribute parallel for` directive is used to specify the offloading region, and the `map(to:GSD)` clause is used to map variables from the host to the device.
