// Randomizes the concentrations of all nuclides in a variety of materials
Kokkos::View<double*> load_concs(Kokkos::View<int*> num_nucs, int max_num_nucs)
{
    uint64_t seed = STARTING_SEED * STARTING_SEED;
    Kokkos::View<double*> concs("concs", 12 * max_num_nucs);

    Kokkos::parallel_for("LoadConcentrations", Kokkos::RangePolicy<>(0, 12), KOKKOS_LAMBDA(int i) {
        for (int j = 0; j < num_nucs(i); j++)
            concs(i * max_num_nucs + j) = LCG_random_double(&seed);
    });

    return concs;
}
