Here is the translated Makefile for the Kokkos execution model:

```makefile
#===============================================================================
# User Options
#===============================================================================

COMPILER    ?= g++
OPTIMIZE    ?= yes
DEBUG       ?= no
PROFILE     ?= no
KOKKOS_PATH ?= /path/to/kokkos # Set this to your Kokkos installation path

#===============================================================================
# Program name & source code list
#===============================================================================

program = XSBench

source = \
Main.cpp \
io.cpp \
Simulation.cpp \
GridInit.cpp \
XSutils.cpp \
Materials.cpp

obj = $(source:.cpp=.o)

#===============================================================================
# Sets Flags
#===============================================================================

# Standard Flags
CFLAGS := 

# Linker Flags
LDFLAGS = -lm

# Kokkos Compiler
ifeq ($(COMPILER),g++)
  CFLAGS += -Wall -O3 -std=c++14 -I$(KOKKOS_PATH)/include
  LDFLAGS += -L$(KOKKOS_PATH)/lib -lkokkos -lpthread -lOpenMP
endif

# Debug Flags
ifeq ($(DEBUG),yes)
  CFLAGS += -g
  LDFLAGS  += -g
endif

# Profiling Flags
ifeq ($(PROFILE),yes)
  CFLAGS += -pg
  LDFLAGS  += -pg
endif

# Optimization Flags
ifeq ($(OPTIMIZE),yes)
  CFLAGS += -O3
endif

#===============================================================================
# Targets to Build
#===============================================================================

$(program): $(obj) XSbench_header.hpp Makefile
	$(CXX) $(CFLAGS) $(obj) -o $@ $(LDFLAGS)

%.o: %.cpp XSbench_header.hpp Makefile
	$(CXX) $(CFLAGS) -c $< -o $@

clean:
	rm -rf $(program) $(obj)

edit:
	vim -p $(source) XSbench_header.hpp

run:
	./$(program)
```

### Notes:
1. Replace `/path/to/kokkos` with the actual path to your Kokkos installation.
2. The file extensions have been changed from `.cu` to `.cpp` to reflect the C++ codebase.
3. The Kokkos library is linked with the appropriate flags for threading and OpenMP support.
4. The Makefile assumes that Kokkos is installed and properly configured on your system. Adjust the flags as necessary based on your specific setup.
