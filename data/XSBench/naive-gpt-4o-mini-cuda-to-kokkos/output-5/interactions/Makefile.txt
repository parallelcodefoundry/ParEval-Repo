```makefile
#===============================================================================
# User Options
#===============================================================================

COMPILER    ?= g++
OPTIMIZE    ?= yes
DEBUG       ?= no
PROFILE     ?= no

#===============================================================================
# Program name & source code list
#===============================================================================

program = XSBench

source = \
Main.cpp \
io.cpp \
Simulation.cpp \
GridInit.cpp \
XSutils.cpp \
Materials.cpp

obj = $(source:.cpp=.o)

#===============================================================================
# Sets Flags
#===============================================================================

# Standard Flags
CFLAGS := 

# Linker Flags
LDFLAGS = -lm

# Kokkos Compiler
ifeq ($(COMPILER),g++)
  CXX = $(COMPILER)
  CFLAGS += -Wall -O3 -std=c++14 -DKOKKOS_ENABLE_OPENMP -I/path/to/kokkos/include
endif

# Debug Flags
ifeq ($(DEBUG),yes)
  CFLAGS += -g
endif

# Profiling Flags
ifeq ($(PROFILE),yes)
  CFLAGS += -pg
endif

# Optimization Flags
ifeq ($(OPTIMIZE),yes)
  CFLAGS += -O3
endif

#===============================================================================
# Targets to Build
#===============================================================================

$(program): $(obj) XSbench_header.hpp Makefile
	$(CXX) $(CFLAGS) $(obj) -o $@ $(LDFLAGS)

%.o: %.cpp XSbench_header.hpp Makefile
	$(CXX) $(CFLAGS) -c $< -o $@

clean:
	rm -rf $(program) $(obj)

edit:
	vim -p $(source) XSbench_header.hpp

run:
	./$(program)
```

### Notes:
1. The Kokkos library is included with the `-I/path/to/kokkos/include` flag. You should replace `/path/to/kokkos/include` with the actual path to your Kokkos installation.
2. The file extensions have been changed from `.cu` to `.cpp` to reflect the C++ codebase.
3. The CUDA-specific flags and commands have been removed, and the Kokkos execution model is enabled with the `-DKOKKOS_ENABLE_OPENMP` flag.
4. Ensure that you have Kokkos installed and properly configured in your environment for this Makefile to work correctly.
