```cmake
cmake_minimum_required(VERSION 3.10)
project(XSBench_Kokkos)

# Find the Kokkos library
find_package(Kokkos REQUIRED)

# Set the C++ standard
set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Set the source files
set(SOURCE
  Main.cpp
  io.cpp
  Simulation.cpp
  GridInit.cpp
  XSutils.cpp
  Materials.cpp
)

# Add the executable
add_executable(XSBench ${SOURCE})

# Target properties
target_include_directories(XSBench PUBLIC ${Kokkos_INCLUDE_DIRS})
target_link_libraries(XSBench ${Kokkos_LIBRARIES})

# Install the executable
install(TARGETS XSBench DESTINATION bin)
```
